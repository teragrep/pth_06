238 private void writeFilter(String tableName, int filterId) {
239 Assertions.assertDoesNotThrow(() -> {
240 conn.prepareStatement("CREATE SCHEMA IF NOT EXISTS BLOOMDB").execute();
241 conn.prepareStatement("USE BLOOMDB").execute();
242 String sql = "INSERT INTO `" + tableName + "` (`partition_id`, `filter_type_id`, `filter`) "
243 + "VALUES (?, (SELECT `id` FROM `filtertype` WHERE id=?), ?)";
244 PreparedStatement stmt = conn.prepareStatement(sql);
245 BloomFilter filter = BloomFilter.create(1000, 0.01);
246 final ByteArrayOutputStream filterBAOS = new ByteArrayOutputStream();
247 Assertions.assertDoesNotThrow(() -> {
248 filter.writeTo(filterBAOS);
249 filterBAOS.close();
250 });
251 stmt.setInt(1, 1);
252 stmt.setInt(2, filterId);
253 stmt.setBytes(3, filterBAOS.toByteArray());
254 stmt.executeUpdate();
255 stmt.close();
