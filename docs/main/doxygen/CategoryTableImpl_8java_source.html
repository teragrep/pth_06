113 public void create() {
114 final Table<Record> namedTable = DSL.table(DSL.name(("term_" + bloomTermId + "_" + originTable.getName())));
115 if (LOGGER.isInfoEnabled()) {
116 LOGGER.info("Creating temporary table <{}>", namedTable.getName());
117 }
118
119 try (final DropTableStep dropTableStep = ctx.dropTemporaryTableIfExists(namedTable)) {
120 dropTableStep.execute();
121 }
122
123 final String sql = "create temporary table " + namedTable.getName()
124 + "(id bigint auto_increment primary key, term_id bigint, type_id bigint, filter longblob, unique key "
125 + namedTable.getName() + "_unique_key (term_id, type_id))";
126 final Query createQuery = ctx.query(sql);
127 createQuery.execute();
128 final Index typeIndex = DSL.index(DSL.name(namedTable.getName() + "_ix_type_id"));
129
130 try (
131 final CreateIndexIncludeStep indexStep = ctx.createIndex(typeIndex).on(namedTable, DSL.field("type_id", BIGINTUNSIGNED.nullable(false)))
132 ) {
133 LOGGER.trace("BloomFilterTempTable create index <{}>", indexStep);
134 indexStep.execute();
135 }
