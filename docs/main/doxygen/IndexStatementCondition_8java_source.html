73 public IndexStatementCondition(String value, ConditionConfig config, Condition condition) {
74 this.value = value;
75 this.config = config;
76 this.condition = condition;
77 this.tableSet = new HashSet<>();
78 }
79
80 public Condition condition() {
81 if (!config.bloomEnabled()) {
82 LOGGER.debug("Indexstatement reached with bloom disabled");
83 return condition;
84 }
85 Condition newCondition = condition;
86 if (tableSet.isEmpty()) {
87 // get all tables that pattern match with search value
88 final QueryCondition tableFilteringCondition = new RegexLikeCondition(value);
89 final DatabaseTables conditionMatchingTables = new ConditionMatchBloomDBTables(
90 config.context(),
91 tableFilteringCondition
92 );
93 tableSet.addAll(conditionMatchingTables.tables());
94 }
95 if (!tableSet.isEmpty()) {
96 if (LOGGER.isDebugEnabled()) {
97 LOGGER.debug("Found pattern match on <{}> table(s)", tableSet.size());
98 }
99 Condition combinedTableCondition = DSL.noCondition();
100 Condition combinedNullFilterCondition = DSL.noCondition();
101
102 for (final Table<?> table : tableSet) {
103 // create a category temp table with filters
104 final CategoryTable categoryTable = new CategoryTableWithFilters(
105 config.context(),
106 table,
107 config.bloomTermId(),
108 value
109 );
110 categoryTable.create();
111 // create table condition for table
112 final Condition nullFilterCondition = table.field("filter").isNull();
113 final QueryCondition tableCondition = new CategoryTableCondition(table, config.bloomTermId());
114 combinedTableCondition = combinedTableCondition.or(tableCondition.condition());
115 combinedNullFilterCondition = combinedNullFilterCondition.and(nullFilterCondition);
116 }
117
118 newCondition = combinedTableCondition.or(combinedNullFilterCondition);
119 }
120 return newCondition;
121 }
