108 public StreamDBClient(Config config) throws SQLException {
109
110 this.bloomEnabled = config.archiveConfig.bloomEnabled;
111
112 LOGGER.info("StreamDBClient bloom.enabled: " + this.bloomEnabled);
113
114 final String userName = config.archiveConfig.dbUsername;
115 final String password = config.archiveConfig.dbPassword;
116 final String url = config.archiveConfig.dbUrl;
117 final String journaldbName = config.archiveConfig.dbJournalDbName;
118 final String streamdbName = config.archiveConfig.dbStreamDbName;
119 final String bloomdbName = config.archiveConfig.bloomDbName;
120 final boolean hideDatabaseExceptions = config.archiveConfig.hideDatabaseExceptions;
121 final boolean withoutFilters = config.archiveConfig.withoutFilters;
122 final String withoutFiltersPattern = config.archiveConfig.withoutFiltersPattern;
123
124 // https://blog.jooq.org/how-i-incorrectly-fetched-jdbc-resultsets-again/
125 Settings settings = new Settings()
126 .withRenderMapping(new RenderMapping().withSchemata(new MappedSchema().withInput("streamdb").withOutput(streamdbName), new MappedSchema().withInput("journaldb").withOutput(journaldbName), new MappedSchema().withInput("bloomdb").withOutput(bloomdbName)));
127 if (hideDatabaseExceptions) {
128 settings = settings.withThrowExceptions(ThrowExceptions.THROW_NONE);
129 LOGGER.warn("StreamDBClient SQL Exceptions set to THROW_NONE");
130 }
131
132 System.getProperties().setProperty("org.jooq.no-logo", "true");
133 final Connection connection = DriverManager.getConnection(url, userName, password);
134 this.ctx = DSL.using(connection, SQLDialect.MYSQL, settings);
135
136 if (hideDatabaseExceptions) {
137 // force sql mode to NO_ENGINE_SUBSTITUTION, STRICT mode
138 ctx.execute("SET sql_mode = 'NO_ENGINE_SUBSTITUTION';");
139 }
140
141 // -- TODO use dslContext.batch for all initial operations
142 final FilterlessSearch filterlessSearch;
143 if (withoutFilters) {
144 filterlessSearch = new FilterlessSearchImpl(ctx, withoutFiltersPattern);
145 }
146 else {
147 filterlessSearch = new FilterlessSearchStub();
148 }
149
150 this.walker = new ConditionWalker(ctx, bloomEnabled, filterlessSearch);
151 Condition streamdbCondition;
152
153 try {
154 // Construct both streamDB and journalDB query conditions
155 streamdbCondition = walker.fromString(config.query, true);
156 this.journaldbCondition = walker.fromString(config.query, false);
157 }
158 catch (ParserConfigurationException | IOException | SAXException e) {
159 throw new IllegalArgumentException(e);
160 }
161
162 GetArchivedObjectsFilterTable.create(ctx, streamdbCondition); // TEMPTABLE
163 SliceTable.create(ctx);
164
165 // by default no cutoff
166 includeBeforeEpoch = Long.MAX_VALUE;
167
168 this.metricRegistry = new MetricRegistry();

185 public int pullToSliceTable(Date day) {
186 NestedTopNQuery nestedTopNQuery = new NestedTopNQuery();
187 SelectOnConditionStep<Record11<ULong, String, String, String, String, Date, String, String, Long, ULong, ULong>> select = ctx
188 .select(
189 JOURNALDB.LOGFILE.ID, nestedTopNQuery.directory, nestedTopNQuery.stream, JOURNALDB.HOST.NAME,
190 JOURNALDB.LOGFILE.LOGTAG, JOURNALDB.LOGFILE.LOGDATE, JOURNALDB.BUCKET.NAME,
191 JOURNALDB.LOGFILE.PATH, nestedTopNQuery.logtime, JOURNALDB.LOGFILE.FILE_SIZE,
192 JOURNALDB.LOGFILE.UNCOMPRESSED_FILE_SIZE
193 )
194 .from(nestedTopNQuery.getTableStatement(journaldbCondition, day))
195 .join(JOURNALDB.LOGFILE)
196 .on(JOURNALDB.LOGFILE.ID.eq(nestedTopNQuery.id))
197 .join(JOURNALDB.BUCKET)
198 .on(JOURNALDB.BUCKET.ID.eq(JOURNALDB.LOGFILE.BUCKET_ID))
199 .join(JOURNALDB.HOST)
200 .on(JOURNALDB.HOST.ID.eq(JOURNALDB.LOGFILE.HOST_ID));
201
202 LOGGER.trace("StreamDBClient.pullToSliceTable select <{}>", select);
203 final Timer.Context timerCtx = metricRegistry.timer("ArchiveDatabaseLatency").time();
204 final int rows;
205
206 try (final InsertOnDuplicateStep<Record> selectStep = ctx.insertInto(SliceTable.SLICE_TABLE).select(select)) {
207 rows = selectStep.execute();
208 }
209
210 final long latencyNs = timerCtx.stop();
211
212 if (rows != 0) {
213 metricRegistry.histogram("ArchiveDatabaseLatencyPerRow").update(latencyNs / rows);
214 }
215
216 LOGGER.info("StreamDBClient.pullToSliceTable" + ": took (" + "<{}> ms)", (latencyNs / 1_000_000d));
217
218 metricRegistry.counter("ArchiveDatabaseRowCount").inc(rows);
219
220 return rows;
221
222 }

224 WeightedOffset getNextHourAndSizeFromSliceTable(long previousHour) {
225 Result<Record2<Long, ULong>> hourAndFilesizeRecord = ctx
226 .selectDistinct(SliceTable.logtime, SliceTable.filesize)
227 .from(SliceTable.SLICE_TABLE)
228 .where(SliceTable.logtime.greaterThan(previousHour).and(SliceTable.logtime.lessThan(includeBeforeEpoch))).orderBy(SliceTable.logtime.asc()).limit(1).fetch();
229
230 final WeightedOffset weightedOffset;
231 if (hourAndFilesizeRecord.isEmpty()) {
232 weightedOffset = new WeightedOffset();
233 }
234 else {
235 long offset = hourAndFilesizeRecord.get(0).get(0, Long.class);
236 long fileSize = hourAndFilesizeRecord.get(0).get(1, ULong.class).longValue();
237
238 weightedOffset = new WeightedOffset(offset, fileSize);
239 }
240
241 return weightedOffset;
242

319 private static void create(DSLContext ctx, Condition streamdbCondition) {
320 DropTableStep dropQuery = ctx.dropTemporaryTableIfExists(GetArchivedObjectsFilterTable.FILTER_TABLE);
321 dropQuery.execute();
322
323 CreateTableWithDataStep query = ctx
324 .createTemporaryTable(GetArchivedObjectsFilterTable.FILTER_TABLE)
325 .as(
326 select(
327 // these are hardcoded for the procedure execution
328 STREAMDB.STREAM.DIRECTORY.as(GetArchivedObjectsFilterTable.directory)
329 )
330 .select(STREAMDB.STREAM.STREAM_.as(GetArchivedObjectsFilterTable.stream))
331 .select(STREAMDB.STREAM.TAG.as(GetArchivedObjectsFilterTable.tag))
332 .select((JOURNALDB.HOST.NAME.as(GetArchivedObjectsFilterTable.host)))
333 .select((JOURNALDB.HOST.ID.as(GetArchivedObjectsFilterTable.host_id)))
334 .from(STREAMDB.STREAM)
335 .innerJoin(STREAMDB.LOG_GROUP)
336 .on((STREAMDB.STREAM.GID).eq(STREAMDB.LOG_GROUP.ID))
337 .innerJoin(STREAMDB.HOST)
338 .on((STREAMDB.HOST.GID).eq(STREAMDB.LOG_GROUP.ID))
339 .innerJoin(JOURNALDB.HOST)
340 .on((STREAMDB.HOST.NAME).eq(JOURNALDB.HOST.NAME))
341 // following change
342 .where(streamdbCondition)
343 );
344 query.execute();
345
346 // this could be within tmpTableCreateSql but JOOQ can't (yet) https://github.com/jOOQ/jOOQ/issues/11752
347 try (
348 final CreateIndexIncludeStep indexStep = ctx.createIndex(GetArchivedObjectsFilterTable.hostIndex)
349 //.on(FILTER_TABLE, directory, host_id, tag, stream).execute(); // FIXME this happens only on dev kube due to old mariadb: Index column size too large. The maximum column size is 767 bytes.
350 .on(
351 GetArchivedObjectsFilterTable.FILTER_TABLE, GetArchivedObjectsFilterTable.host_id,
352 GetArchivedObjectsFilterTable.tag
353 )
354 ) {
355 indexStep.execute();

361 public class NestedTopNQuery {
362
363 private final String innerTableName = "limited";
364 private final Table<Record> innerTable = DSL.table(DSL.name(innerTableName));
365
366 // TODO refactor: heavily database session dependant: create synthetic logtime field, based on the path
367 public final Field<Long> logtimeFunction = DSL
368 .field(
369 "UNIX_TIMESTAMP(STR_TO_DATE(SUBSTRING(REGEXP_SUBSTR({0},'^\\\\d{4}\\\\/\\\\d{2}-\\\\d{2}\\\\/[\\\\w\\\\.-]+\\\\/([^\\\\p{Z}\\\\p{C}]+?)\\\\/([^\\\\p{Z}\\\\p{C}]+)(-@)?(\\\\d+|)-(\\\\d{4}\\\\d{2}\\\\d{2}\\\\d{2})'), -10, 10), '%Y%m%d%H'))",
370 Long.class, JOURNALDB.LOGFILE.PATH
371 );
372
373 private final Field<ULong> id = DSL.field(DSL.name(innerTableName, "id"), ULong.class);
374 private final Field<String> directory = DSL.field(DSL.name(innerTableName, "directory"), String.class);
375 private final Field<String> stream = DSL.field(DSL.name(innerTableName, "stream"), String.class);
376 private final Field<Long> logtime = DSL.field(DSL.name(innerTableName, "logtime"), Long.class);
377 private final Field<Long> logtimeForOrderBy = DSL.field("logtime", Long.class);
378
379 private final SelectField<?>[] resultFields = {
380 JOURNALDB.LOGFILE.ID.as(id),
381 GetArchivedObjectsFilterTable.directory.as(directory),
382 GetArchivedObjectsFilterTable.stream.as(stream),
383 logtimeFunction.as(logtime)
384 };
385
386 private Table<Record> getTableStatement(Condition journaldbConditionArg, Date day) {
387
388 SelectOnConditionStep<Record> selectOnConditionStep = select(resultFields)
389 .from(GetArchivedObjectsFilterTable.FILTER_TABLE)
390 .innerJoin(JOURNALDB.LOGFILE)
391 .on(JOURNALDB.LOGFILE.HOST_ID.eq(GetArchivedObjectsFilterTable.host_id).and(JOURNALDB.LOGFILE.LOGTAG.eq(GetArchivedObjectsFilterTable.tag)));
392
393 if (bloomEnabled) {
394 // join all tables needed for the condition generated by walker
395 final Set<Table<?>> tables = walker.conditionRequiredTables();
396 if (!tables.isEmpty()) {
397 for (final Table<?> table : tables) {
398 if (LOGGER.isInfoEnabled()) {
399 LOGGER.info("Left join pattern match table: <{}>", table.getName());
400 }
401 selectOnConditionStep = selectOnConditionStep
402 .leftJoin(table)
403 .on(JOURNALDB.LOGFILE.ID.eq((Field<ULong>) table.field("partition_id")));
404 }
405 }
406 }
407
408 return selectOnConditionStep
409 .where(JOURNALDB.LOGFILE.LOGDATE.eq(day).and(journaldbConditionArg))
410 .orderBy(logtimeForOrderBy, JOURNALDB.LOGFILE.ID.asc())
411 .asTable(innerTable);
412 }
413 }
