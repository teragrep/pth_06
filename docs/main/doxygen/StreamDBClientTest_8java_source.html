114 private LogfileRecord logfileRecordForEpoch(long epoch, boolean hasNullEpochColumns) {
115 Instant instant = Instant.ofEpochSecond(epoch);
116 ZonedDateTime zonedDateTime = instant.atZone(zoneId); // expects path dates to be in same timezone as mariadb system timezone
117 int year = zonedDateTime.getYear();
118 // format 0 in front of string if 1-9
119 String month = String.format("%02d", zonedDateTime.getMonthValue());
120 String day = String.format("%02d", zonedDateTime.getDayOfMonth());
121 String hour = String.format("%02d", zonedDateTime.getHour());
122
123 String filename = "example.log-@" + epoch + "-" + year + month + day + hour + ".log.gz";
124 String path = year + "/" + month + "-" + day + "/example.tg.dev.test/example/" + filename;
125 System.out.println("path: " + path);
126 LogfileRecord logfileRecord = new LogfileRecord(
127 ULong.valueOf(epoch),
128 Date.valueOf(zonedDateTime.toLocalDate()),
129 Date.valueOf(zonedDateTime.plusYears(1).toLocalDate()),
130 UShort.valueOf(1),
131 path,
132 null,
133 UShort.valueOf(1),
134 filename,
135 new Timestamp(epoch),
136 ULong.valueOf(120L),
137 "sha256 checksum 1",
138 "archive tag 1",
139 "example",
140 UShort.valueOf(2),
141 UShort.valueOf(1),
142 ULong.valueOf(390L),
143 ULong.valueOf(epoch),
144 ULong.valueOf(epoch + (365 * 24 * 3600)),
145 ULong.valueOf(epoch),
146 ULong.valueOf(1)
147 );
148
149 LogfileRecord nullEpochRecord = new LogfileRecord(
150 ULong.valueOf(epoch),
151 Date.valueOf(zonedDateTime.toLocalDate()),
152 Date.valueOf(zonedDateTime.plusYears(1).toLocalDate()),
153 UShort.valueOf(1),
154 path,
155 null,
156 UShort.valueOf(1),
157 filename,
158 new Timestamp(epoch),
159 ULong.valueOf(120L),
160 "sha256 checksum 1",
161 "archive tag 1",
162 "example",
163 UShort.valueOf(2),
164 UShort.valueOf(1),
165 ULong.valueOf(390L),
166 null,
167 null,
168 null,
169 ULong.valueOf(1)
170 );
171
172 if (hasNullEpochColumns) {
173 return nullEpochRecord;
174 }
175 return logfileRecord;
176 }

302 @Test
303 public void weightedOffsetTest() {
304 // Add test data to logfile table in journaldb.
305 final DSLContext ctx = DSL.using(connection, SQLDialect.MYSQL);
306 final Instant instant = Instant.ofEpochSecond(1696471200L);
307
308 final ZonedDateTime baseTime = ZonedDateTime.ofInstant(instant, zoneId);
309 final ZonedDateTime baseMinusOneHour = baseTime.minusHours(1);
310 final ZonedDateTime basePlusOneHour = baseTime.plusMinutes(1);
311 final ZonedDateTime basePlusOneDay = baseTime.plusDays(1);
312
313 final LogfileRecord baseRecord = logfileRecordForEpoch(baseTime.toEpochSecond(), true);
314 ctx.insertInto(JOURNALDB.LOGFILE).set(baseRecord).execute();
315 final LogfileRecord plusOneHourRecord = logfileRecordForEpoch(basePlusOneHour.toEpochSecond(), true);
316 ctx.insertInto(JOURNALDB.LOGFILE).set(plusOneHourRecord).execute();
317 final LogfileRecord plusOneDayRecord = logfileRecordForEpoch(basePlusOneDay.toEpochSecond(), true);
318 ctx.insertInto(JOURNALDB.LOGFILE).set(plusOneDayRecord).execute();
319
320 // Assert StreamDBClient methods work as expected with the test data.
321 final Map<String, String> opts = this.opts;
322 opts.put("DBurl", mariadb.getJdbcUrl());
323
324 final Config config = new Config(opts);
325 final StreamDBClient sdc = Assertions.assertDoesNotThrow(() -> new StreamDBClient(config));
326
327 // pull baseTime to SliceTable and assert weightedOffset to contain filesize=240 (2 rows) for that hour
328 final int baseTimeRows = sdc.pullToSliceTable(Date.valueOf(baseTime.toLocalDate()));
329 Assertions.assertEquals(2, baseTimeRows);
330 final WeightedOffset weightedOffsetForBaseTime = sdc
331 .getNextHourAndSizeFromSliceTable(baseMinusOneHour.toEpochSecond());
332
333 Assertions.assertEquals(baseTime.toEpochSecond(), weightedOffsetForBaseTime.offset());
334 Assertions.assertEquals(240L, weightedOffsetForBaseTime.fileSize());
335
336 // pull baseTime+1day to SliceTable and assert weightedOffset to contain filesize=120 (1 row) for that hour
337 final int plusOneDayRows = sdc.pullToSliceTable(Date.valueOf(basePlusOneDay.toLocalDate()));
338 Assertions.assertEquals(1, plusOneDayRows);
339 final WeightedOffset weightedOffsetForPlusOneDay = sdc
340 .getNextHourAndSizeFromSliceTable(baseTime.toEpochSecond());
341 Assertions.assertEquals(basePlusOneDay.toEpochSecond(), weightedOffsetForPlusOneDay.offset());
342 Assertions.assertEquals(120L, weightedOffsetForPlusOneDay.fileSize());
343 }
