137 @Override
138 public Offset initialOffset() {
139 // archive only: subtract 3600s (1 hour) from earliest to return first row (start exclusive)
140 DatasourceOffset rv;
141 if (this.config.isArchiveEnabled && !this.config.isKafkaEnabled) {
142 // only archive
143 rv = new DatasourceOffset(new LongOffset(this.aq.getInitialOffset() - 3600L));
144 }
145 else if (!this.config.isArchiveEnabled && this.config.isKafkaEnabled) {
146 // only kafka
147 rv = new DatasourceOffset(new KafkaOffset(this.kq.getBeginningOffsets(null)));
148 }
149 else if (this.config.isArchiveEnabled) {
150 // both
151 rv = new DatasourceOffset(
152 new LongOffset(this.aq.getInitialOffset() - 3600L),
153 new KafkaOffset(this.kq.getBeginningOffsets(null))
154 );
155 }
156 else {
157 // neither
158 throw new IllegalStateException("no datasources enabled, can't get initial offset");
159 }
160 LOGGER.debug("offset[initial]= {}", rv);
161 return rv;
162 }

223 @Override
224 public InputPartition[] planInputPartitions(Offset start, Offset end) {
225 List<InputPartition> inputPartitions = new LinkedList<>();
226
227 Batch currentBatch = new Batch(config, aq, kq).processRange(start, end);
228
229 for (LinkedList<BatchSlice> taskObjectList : currentBatch) {
230
231 // archive tasks
232 LinkedList<ArchiveS3ObjectMetadata> archiveTaskList = new LinkedList<>();
233 for (BatchSlice batchSlice : taskObjectList) {
234 if (batchSlice.type.equals(BatchSlice.Type.ARCHIVE)) {
235 archiveTaskList.add(batchSlice.archiveS3ObjectMetadata);
236 }
237 }
238
239 if (!archiveTaskList.isEmpty()) {
240 inputPartitions
241 .add(
242 new ArchiveMicroBatchInputPartition(
243 config.archiveConfig.s3EndPoint,
244 config.archiveConfig.s3Identity,
245 config.archiveConfig.s3Credential,
246 archiveTaskList,
247 config.auditConfig.query,
248 config.auditConfig.reason,
249 config.auditConfig.user,
250 config.auditConfig.pluginClassName,
251 config.archiveConfig.skipNonRFC5424Files
252 )
253 );
254 }
255
256 // kafka tasks
257 for (BatchSlice batchSlice : taskObjectList) {
258 if (batchSlice.type.equals(BatchSlice.Type.KAFKA)) {
259 inputPartitions
260 .add(
261 new KafkaMicroBatchInputPartition(
262 config.kafkaConfig.executorOpts,
263 batchSlice.kafkaTopicPartitionOffsetMetadata.topicPartition,
264 batchSlice.kafkaTopicPartitionOffsetMetadata.startOffset,
265 batchSlice.kafkaTopicPartitionOffsetMetadata.endOffset,
266 config.kafkaConfig.executorConfig,
267 config.kafkaConfig.skipNonRFC5424Records
268 )
269 );
270 }
271 }
272 }
273
274 return inputPartitions.toArray(new InputPartition[0]);
275 }

296 else {
297 throw new IllegalStateException("No datasources enabled, can't get last used offset");
298 }
