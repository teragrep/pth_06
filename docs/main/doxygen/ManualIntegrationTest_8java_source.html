227 private boolean isArchiveDone(StreamingQuery outQ) {
228 Boolean archiveDone = true;
229 for (int i = 0; i < outQ.lastProgress().sources().length; i++) {
230 String startOffset = outQ.lastProgress().sources()[i].startOffset();
231 String endOffset = outQ.lastProgress().sources()[i].endOffset();
232 String description = outQ.lastProgress().sources()[i].description();
233
234 if (description != null && !description.startsWith("com.teragrep.pth06.ArchiveMicroBatchReader@")) {
235 // ignore others than archive
236 continue;
237 }
238
239 if (startOffset != null) {
240 if (!startOffset.equalsIgnoreCase(endOffset)) {
241 archiveDone = false;
242 }
243 }
244 else {
245 archiveDone = false;
246 }
247 }
248 LOGGER.debug("isArchiveDone(): rv: " + archiveDone);
249 return archiveDone;
