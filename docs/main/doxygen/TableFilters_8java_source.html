98 public Batch asBatch() {
99 final List<InsertValuesStepN<?>> insertValuesStepNList = new ArrayList<>();
100 final Result<Record> result = recordsInMetadata.toResult();
101 for (final Record record : result) {
102 final Field<?>[] insertFields = {
103 DSL.field(DSL.name(thisTable.getName(), "term_id"), BIGINTUNSIGNED.nullable(false)),
104 DSL.field(DSL.name(thisTable.getName(), "type_id"), BIGINTUNSIGNED.nullable(false)),
105 DSL.field(DSL.name(thisTable.getName(), "filter"), byte[].class)
106 };
107 final ULong expectedField = record
108 .getValue(DSL.field(DSL.name(table.getName(), "expectedElements"), ULong.class));
109 final Double fpp = record.getValue(DSL.field(DSL.name(table.getName(), "targetFpp"), Double.class));
110 final String pattern = record.getValue(BLOOMDB.FILTERTYPE.PATTERN, String.class);
111 final SearchTermBloomFilter filter;
112 if (pattern == null) {
113 filter = new SearchTermBloomFilter(expectedField.longValue(), fpp, new TokenizedValue(searchTerm));
114 }
115 else {
116 filter = new SearchTermBloomFilter(
117 expectedField.longValue(),
118 fpp,
119 new RegexExtractedValue(searchTerm, pattern)
120 );
121 }
122 final Field<?>[] valueFields = {
123 DSL.val(bloomTermId, ULong.class),
124 DSL.val(record.getValue(BLOOMDB.FILTERTYPE.ID), ULong.class),
125 DSL.val(filter.bytes(), byte[].class)
126 };
127 final InsertValuesStepN<?> insertStep = ctx.insertInto(thisTable).columns(insertFields).values(valueFields);
128 insertValuesStepNList.add(insertStep);
129 }
130 return ctx.batch(insertValuesStepNList);
131 }
