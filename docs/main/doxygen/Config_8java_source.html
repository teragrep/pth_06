75 public Config(Map<String, String> opts) {
76 this.query = opts.get("queryXML");
77 if (this.query == null) {
78 throw new IllegalArgumentException("no queryXML provided");
79 }
80
81 batchConfig = new BatchConfig(opts);
82 auditConfig = new AuditConfig(opts);
83
84 isArchiveEnabled = opts.getOrDefault("archive.enabled", "false").equalsIgnoreCase("true");
85 if (isArchiveEnabled) {
86 archiveConfig = new ArchiveConfig(opts);
87 }
88 else {
89 archiveConfig = new ArchiveConfig();
90 }
91
92 isKafkaEnabled = opts.getOrDefault("kafka.enabled", "false").equalsIgnoreCase("true");
93 if (isKafkaEnabled) {
94 kafkaConfig = new KafkaConfig(opts);
95 }
96 else {
97 kafkaConfig = new KafkaConfig();
98 }
99
100 // check that at least one datasource is enabled
101 if (!isArchiveEnabled && !isKafkaEnabled) {
102 throw new IllegalStateException("No datasources enabled");
103 }
104
105 // fetch metadata (defaults to false)
106 isMetadataQuery = opts.getOrDefault("metadataQuery.enabled", "false").equalsIgnoreCase("true");
107
108 loggingConfig = new LoggingConfigImpl(opts);
109
110 sqlConfig = new SQLConfigImpl(opts);
111 }
