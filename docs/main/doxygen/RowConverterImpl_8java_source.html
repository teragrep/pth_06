163 @Override
164 public void open() throws IOException {
165 // Open s3-file
166
167 String bucketName = this.bucket;
168 String keyName = this.path;
169
170 String logName = bucketName + "/" + keyName;
171 S3Object s3object;
172 try {
173 LOGGER.debug("Attempting to open file: " + logName);
174 s3object = s3client.getObject(bucketName, keyName);
175
176 if (LOGGER.isDebugEnabled()) {
177 LOGGER
178 .debug(
179 "Open S3 stream bucket:" + bucketName + " keyname=" + keyName + " Metadata length:"
180 + s3object.getObjectMetadata().getContentLength()
181 );
182 }
183 this.inputStream = new BufferedInputStream(s3object.getObjectContent(), 8 * 1024 * 1024);
184 GZIPInputStream gz = new GZIPInputStream(inputStream);
185 rfc5424Frame.load(gz);
186
187 // Set up result
188 LOGGER.trace("S3FileHandler.open() Initialized result set with element lists");
189
190 LOGGER.info("S3FileHandler.open() Initialized parser for: " + logName);
191 }
192 catch (AmazonServiceException amazonServiceException) {
193
194 if (403 == amazonServiceException.getStatusCode()) {
195 LOGGER
196 .error(
197 "Skipping file " + logName + " due to errorCode:"
198 + amazonServiceException.getStatusCode()
199 );
200 }
201 else {
202 throw amazonServiceException;
203 }
204 }

209 public boolean next() throws IOException {
210 if (LOGGER.isDebugEnabled()) {

212 }

218 boolean rv;

220 rv = rfc5424Frame.next();
221 if (rv) {
222 this.currentOffset++;
223
224 if (LOGGER.isDebugEnabled()) {
225 LOGGER.debug("<{}> Read event: <[{}]>", currentOffset, rfc5424Frame);
226 }
227 }
228 }
229 catch (ParseException parseException) {
230 LOGGER.error("ParseException at object: <[{}]>/<[{}]>", bucket, path);
231 if (skipNonRFC5424Files) {
232 rv = false;
233 }
234 else {
235 throw parseException;
236 }
237 }
238 return rv;

241 static long rfc3339ToEpoch(ZonedDateTime zonedDateTime) {
242 final Instant instant = zonedDateTime.toInstant();
243
244 final long MICROS_PER_SECOND = 1000L * 1000L;
245 final long NANOS_PER_MICROS = 1000L;
246 final long sec = Math.multiplyExact(instant.getEpochSecond(), MICROS_PER_SECOND);
247
248 return Math.addExact(sec, instant.getNano() / NANOS_PER_MICROS);
249 }

251 @Override
252 public InternalRow get() {
253 //System.out.println("RowConverter.get>");
254 if (LOGGER.isDebugEnabled()) {
255 LOGGER
256 .debug(
257 "RowConverter.get> Partition (" + this.id + "):" + bucket + "/" + path + " Get("
258 + currentOffset + ")"
259 );
260 }
261
262 if (LOGGER.isDebugEnabled()) {
263 LOGGER.debug("Parsersyslog event:" + rfc5424Frame.toString());
264 }
265
266 final long timeStamp = rfc3339ToEpoch(new RFC5424Timestamp(rfc5424Frame.timestamp).toZonedDateTime());
267
268 // input
269 final byte[] source = eventToSource();
270
271 // origin
272 final byte[] origin = eventToOrigin();
273
274 if (LOGGER.isDebugEnabled()) {
275 LOGGER
276 .trace(
277 "PARSED timestamp: " + timeStamp + " message: "
278 + new String(rfc5424Frame.msg.toBytes(), StandardCharsets.UTF_8)
279 );
280 }
281
282 rowWriter.reset();
283 rowWriter.zeroOutNullBytes();
284 rowWriter.write(0, timeStamp);
285 rowWriter.write(1, UTF8String.fromBytes(rfc5424Frame.msg.toBytes()));
286 rowWriter.write(2, this.directory);
287 rowWriter.write(3, this.stream);
288 rowWriter.write(4, this.host);
289 rowWriter.write(5, UTF8String.fromBytes(source));
290 rowWriter.write(6, this.id);
291 rowWriter.write(7, currentOffset);
292 rowWriter.write(8, UTF8String.fromBytes(origin));
293
294 if (LOGGER.isDebugEnabled()) {
295 LOGGER.debug("Get Event, row=written");
296 }
297
298 auditPlugin
299 .audit(
300 timeStamp, rfc5424Frame.msg.toBytes(), this.directory.getBytes(), this.stream.getBytes(),
301 this.host.getBytes(), source, this.id.toString(), currentOffset
302 );
303 return rowWriter.getRow();
304
305 }

307 private byte[] eventToOrigin() {
308 byte[] origin;
309 Fragment originFragment = rfc5424Frame.structuredData.getValue(originHostname);
310 if (!originFragment.isStub) {
311 origin = originFragment.toBytes();
312 }
313 else {
314 origin = new byte[] {};
315 }
316 return origin;
317 }

319 private byte[] eventToSource() {
320 //input is produced from SD element event_node_source@48577 by
321 // concatenating "source_module:hostname:source". in case
322 //if event_node_source@48577 is not available use event_node_relay@48577.
323 //If neither are present, use null value.
324
325 sourceConcatenationBuffer.clear();
326
327 Fragment sourceModuleFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceSourceModule);
328 if (sourceModuleFragment.isStub) {
329 sourceModuleFragment = rfc5424Frame.structuredData.getValue(eventNodeRelaySourceModule);
330 }
331
332 byte[] source_module;
333 if (!sourceModuleFragment.isStub) {
334 source_module = sourceModuleFragment.toBytes();
335 }
336 else {
337 source_module = new byte[] {};
338 }
339
340 Fragment sourceHostnameFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceHostname);
341 if (sourceHostnameFragment.isStub) {
342 sourceHostnameFragment = rfc5424Frame.structuredData.getValue(eventNodeRelayHostname);
343 }
344
345 byte[] source_hostname;
346 if (!sourceHostnameFragment.isStub) {
347 source_hostname = sourceHostnameFragment.toBytes();
348 }
349 else {
350 source_hostname = new byte[] {};
351 }
352
353 Fragment sourceSourceFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceSource);
354 if (sourceSourceFragment.isStub) {
355 sourceSourceFragment = rfc5424Frame.structuredData.getValue(eventNodeRelaySource);
356 }
357
358 byte[] source_source;
359 if (!sourceSourceFragment.isStub) {
360 source_source = sourceSourceFragment.toBytes();
361 }
362 else {
363 source_source = new byte[] {};
364 }
365
366 // source_module:hostname:source"
367 sourceConcatenationBuffer.put(source_module);
368 sourceConcatenationBuffer.put((byte) ':');
369 sourceConcatenationBuffer.put(source_hostname);
370 sourceConcatenationBuffer.put((byte) ':');
371 sourceConcatenationBuffer.put(source_source);
372
373 sourceConcatenationBuffer.flip();
374 byte[] input = new byte[sourceConcatenationBuffer.remaining()];
375 sourceConcatenationBuffer.get(input);
376
377 return input;
378 }
