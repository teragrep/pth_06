254 private boolean isArchiveDone(StreamingQuery outQ) {
255 boolean archiveDone = true;
256 for (int i = 0; i < outQ.lastProgress().sources().length; i++) {
257 String startOffset = outQ.lastProgress().sources()[i].startOffset();
258 String endOffset = outQ.lastProgress().sources()[i].endOffset();
259 String description = outQ.lastProgress().sources()[i].description();
260
261 if (description != null && !description.startsWith("com.teragrep.pth_06.ArchiveMicroStreamReader@")) {
262 // ignore others than archive
263 continue;
264 }
265
266 if (startOffset != null) {
267 if (!startOffset.equalsIgnoreCase(endOffset)) {
268 archiveDone = false;
269 }
270 }
271 else {
272 archiveDone = false;
273 }
274 }
275 return archiveDone;
276 }

283 private long preloadS3Data() throws IOException {
284 long rows = 0L;
285 AmazonS3 amazonS3 = new Pth06S3Client(s3endpoint, s3identity, s3credential).build();
286
287 TreeMap<Long, Result<Record11<ULong, String, String, String, String, Date, String, String, Long, ULong, ULong>>> virtualDatabaseMap = mockDBData
288 .getVirtualDatabaseMap();
289
290 for (
291 Map.Entry<Long, Result<Record11<ULong, String, String, String, String, Date, String, String, Long, ULong, ULong>>> entry : virtualDatabaseMap
292 .entrySet()
293 ) {
294 Iterator<Record11<ULong, String, String, String, String, Date, String, String, Long, ULong, ULong>> it = entry
295 .getValue()
296 .iterator();
297 while (it.hasNext()) {
298 // id, directory, stream, host, logtag, logdate, bucket, path, logtime, filesize
299 Record11<ULong, String, String, String, String, Date, String, String, Long, ULong, ULong> record10 = it
300 .next();
301 Long id = record10.get(0, ULong.class).longValue();
302 String directory = record10.get(1, String.class);
303 String stream = record10.get(2, String.class);
304 String host = record10.get(3, String.class);
305 String logtag = record10.get(4, String.class);
306 Date logdate = record10.get(5, Date.class);
307 String bucket = record10.get(6, String.class);
308 String path = record10.get(7, String.class);
309 Long logtime = record10.get(8, ULong.class).longValue();
310 Long filesize = record10.get(9, ULong.class).longValue();
311
312 String recordAsJson = record10.formatJSON();
313
314 // <46>1 2010-01-01T12:34:56.123456+02:00 hostname.domain.tld pstats - -
315 SyslogMessage syslog = new SyslogMessage();
316 syslog = syslog
317 .withFacility(Facility.USER)
318 .withSeverity(Severity.WARNING)
319 .withTimestamp(logtime)
320 .withHostname(host)
321 .withAppName(logtag)
322 .withMsg(recordAsJson);
323
324 // [event_id@48577 hostname="hostname.domain.tld" uuid="" unixtime="" id_source="source"]
325
326 SDElement event_id_48577 = new SDElement("event_id@48577")
327 .addSDParam("hostname", host)
328 .addSDParam("uuid", UUID.randomUUID().toString())
329 .addSDParam("source", "source")
330 .addSDParam("unixtime", Long.toString(System.currentTimeMillis()));
331
332 syslog = syslog.withSDElement(event_id_48577);
333
334 // [event_format@48577 original_format="rfc5424"]
335
336 SDElement event_format_48577 = new SDElement("event_id@48577").addSDParam("original_format", "rfc5424");
337
338 syslog = syslog.withSDElement(event_format_48577);
339
340 // [event_node_relay@48577 hostname="relay.domain.tld" source="hostname.domain.tld" source_module="imudp"]
341
342 SDElement event_node_relay_48577 = new SDElement("event_node_relay@48577")
343 .addSDParam("hostname", "relay.domain.tld")
344 .addSDParam("source", host)
345 .addSDParam("source_module", "imudp");
346
347 syslog = syslog.withSDElement(event_node_relay_48577);
348
349 // [event_version@48577 major="2" minor="2" hostname="relay.domain.tld" version_source="relay"]
350
351 SDElement event_version_48577 = new SDElement("event_version@48577")
352 .addSDParam("major", "2")
353 .addSDParam("minor", "2")
354 .addSDParam("hostname", "relay.domain.tld")
355 .addSDParam("version_source", "relay");
356
357 syslog = syslog.withSDElement(event_version_48577);
358
359 // [event_node_router@48577 source="relay.domain.tld" source_module="imrelp" hostname="router.domain.tld"]
360
361 SDElement event_node_router_48577 = new SDElement("event_node_router@48577")
362 .addSDParam("source", "relay.domain.tld")
363 .addSDParam("source_module", "imrelp")
364 .addSDParam("hostname", "router.domain.tld");
365
366 syslog = syslog.withSDElement(event_node_router_48577);
367
368 // [origin@48577 hostname="original.hostname.domain.tld"]
369
370 SDElement origin_48577 = new SDElement("origin@48577")
371 .addSDParam("hostname", "original.hostname.domain.tld");
372 syslog = syslog.withSDElement(origin_48577);
373
374 // check if this bucket exists
375 boolean bucketExists = false;
376 for (Bucket existingBucket : amazonS3.listBuckets()) {
377 if (existingBucket.getName().equals(bucket)) {
378 bucketExists = true;
379 break;
380 }
381 }
382 if (!bucketExists) {
383 amazonS3.createBucket(bucket);
384 }
385
386 // compress the message
387 String syslogMessage = syslog.toRfc5424SyslogMessage();
388
389 ByteArrayOutputStream outStream = new ByteArrayOutputStream(syslogMessage.length());
390 GZIPOutputStream gzip = new GZIPOutputStream(outStream);
391 gzip.write(syslogMessage.getBytes());
392 gzip.close();
393
394 ByteArrayInputStream inStream = new ByteArrayInputStream(outStream.toByteArray());
395
396 // upload as file
397 amazonS3.putObject(bucket, path, inStream, null);
398 rows++;
399 }
400
401 }
402 return rows;
