201 .option("TeragrepAuditUser", System.getProperty("user.name"))

203 .option("kafka.enabled", "false")
204 .option("kafka.bootstrap.servers", "")
205 .option("kafka.sasl.mechanism", "")
206 .option("kafka.security.protocol", "")

212 .load();
213
214 StreamingQuery sq = df.writeStream().foreachBatch((ds, i) -> {
215 ds
216 .select("partition", "_raw")
217 .collectAsList()
218 .forEach(r -> partitionToUncompressedMapping.put(r.getAs(0), r.getAs(1)));
219 }).start();
220 sq.processAllAvailable();
221 sq.stop();
222 sq.awaitTermination();
223
224 int loops = 0;
225 for (Map.Entry<String, String> entry : partitionToUncompressedMapping.entrySet()) {
226 assertFalse(entry.getValue().isEmpty());
227 JsonObject jo = new Gson().fromJson(entry.getValue(), JsonObject.class);
228 assertTrue(jo.has("compressed"));
229 assertTrue(jo.has("uncompressed"));
230 loops++;
231 }
232 Assertions.assertEquals(33, loops);
233 //partition=19181 has NULL for uncompressed size
234 assertEquals(
235 -1L, new Gson().fromJson(partitionToUncompressedMapping.get("19181"), JsonObject.class).get("uncompressed").getAsLong()
236 );
237 }

239 private boolean isArchiveDone(StreamingQuery outQ) {
240 boolean archiveDone = true;
241 for (int i = 0; i < outQ.lastProgress().sources().length; i++) {
242 String startOffset = outQ.lastProgress().sources()[i].startOffset();
243 String endOffset = outQ.lastProgress().sources()[i].endOffset();
244 String description = outQ.lastProgress().sources()[i].description();
245
246 if (description != null && !description.startsWith("com.teragrep.pth_06.ArchiveMicroStreamReader@")) {
247 // ignore others than archive
248 continue;
249 }
250
251 if (startOffset != null) {
252 if (!startOffset.equalsIgnoreCase(endOffset)) {
253 archiveDone = false;
254 }
255 }
256 else {
257 archiveDone = false;
258 }
259 }
260 return archiveDone;
261 }
