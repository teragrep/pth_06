86 public ZonedDateTime toZonedDateTime() {
87 final ZonedDateTime zonedDateTime;
88 if (hasHourlyData()) {
89 zonedDateTime = deriveFromHour();
90 }
91 else if (hasDateData()) {
92 zonedDateTime = deriveFromDate();
93 }
94 else {
95 throw new IllegalStateException("Path does not contain date information: <" + path + ">");
96 }
97 return zonedDateTime;
98 }
99
100 public boolean hasHourlyData() {
101 return hourPattern.matcher(path).matches();
102 }
103
104 public boolean hasDateData() {
105 return datePattern.matcher(path).matches();
106 }

108 private ZonedDateTime deriveFromHour() {
109 final Matcher matcher = hourPattern.matcher(path);
110 if (!matcher.matches()) {
111 throw new IllegalStateException("Hour pattern did not match: <" + path + ">");
112 }
113 final int year = Integer.parseInt(matcher.group("year"));
114 final int month = Integer.parseInt(matcher.group("month"));
115 final int day = Integer.parseInt(matcher.group("day"));
116 final int hour = Integer.parseInt(matcher.group("hour"));
117 return ZonedDateTime.of(year, month, day, hour, 0, 0, 0, zoneId);
118 }
119
120 private ZonedDateTime deriveFromDate() {
121 final Matcher matcher = datePattern.matcher(path);
122 if (!matcher.matches()) {
123 throw new IllegalStateException("Date pattern did not match: <" + path + ">");
124 }
125 final int year = Integer.parseInt(matcher.group("year"));
126 final int month = Integer.parseInt(matcher.group("month"));
127 final int day = Integer.parseInt(matcher.group("day"));
128 return ZonedDateTime.of(year, month, day, 0, 0, 0, 0, zoneId);
129 }
