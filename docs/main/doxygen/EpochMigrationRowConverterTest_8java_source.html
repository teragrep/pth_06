174 private void load(final String content) throws IOException {
175 final ByteArrayOutputStream baos = new ByteArrayOutputStream();
176 try (final GZIPOutputStream gzip = new GZIPOutputStream(baos)) {
177 gzip.write(content.getBytes(StandardCharsets.UTF_8));
178 }
179 final ByteArrayInputStream inStream = new ByteArrayInputStream(baos.toByteArray());
180 amazonS3.putObject(bucket, path, inStream, null);
181 }
182
183 private void ensureBucketExists() {
184 boolean bucketExists = false;
185 for (final Bucket existingBucket : amazonS3.listBuckets()) {
186 if (existingBucket.getName().equals(bucket)) {
187 bucketExists = true;
188 break;
189 }
190 }
191 if (!bucketExists) {
192 amazonS3.createBucket(bucket);
193 }
194 }
