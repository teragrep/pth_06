126 @Override
127 public void open() throws IOException {
128 // Open s3-file
129 final String logName = bucket + "/" + path;
130 final S3Object s3object;
131 try {
132 LOGGER.debug("Attempting to open file <[{}]>", logName);
133 s3object = s3client.getObject(bucket, path);
134 if (LOGGER.isDebugEnabled()) {
135 LOGGER
136 .debug(
137 "Open S3 stream bucket <[{}]> keyname <[{}]> Metadata length <{}>", bucket, path,
138 s3object.getObjectMetadata().getContentLength()
139 );
140 }
141 inputStream = new BufferedInputStream(s3object.getObjectContent(), 8 * 1024 * 1024);
142 final GZIPInputStream gz = new GZIPInputStream(inputStream);
143 rfc5424Frame.load(gz);
144 LOGGER.trace("S3FileHandler.open() Initialized result set with element lists");
145 LOGGER.info("S3FileHandler.open() Initialized parser for <[{}]>", logName);
146 }
147 catch (final AmazonServiceException amazonServiceException) {
148 if (403 == amazonServiceException.getStatusCode()) {
149 LOGGER.error("Skipping file <[{}]> due to errorCode <{}>", logName, 403);
150 }
151 else {
152 throw amazonServiceException;
153 }

158 @Override
159 public boolean next() throws IOException {
160 if (LOGGER.isDebugEnabled()) {
161 LOGGER.debug("RowConverter.next() called, read attempted before <{}> ", readAttempted);
162 }
163 if (LOGGER.isTraceEnabled()) {
164 LOGGER.trace("Read event from S3-file: <[{}]>/<[{}]>", bucket, path);

166
167 boolean hasRow;
168 if (!readAttempted) {
169 readAttempted = true;
170 try {
171 isSyslogFormat = rfc5424Frame.next();
172 }
173 catch (final ParseException exception) {
174 LOGGER
175 .error(
176 "ParseException at object: <[{}]>/<[{}]> - exception message: <{}>", bucket, path,
177 exception.getMessage()
178 );
179 isSyslogFormat = false;
180 }
181 hasRow = true;
182 }
183 else {
184 hasRow = false;
185 }
186 return hasRow;

189 @Override
190 public InternalRow get() {
191 if (LOGGER.isDebugEnabled()) {
192 LOGGER
193 .debug(
194 "EpochMigrationRowConverter.get() partition=<[{}]>, bucket=<[{}]> path=<[{}]> read attempted=<{}>",
195 id, bucket, path, readAttempted
196 );
197 LOGGER.debug("Parser syslog event <[{}]>", rfc5424Frame.toString());
198 }
199
200 rowWriter.reset();
201 rowWriter.zeroOutNullBytes();
202
203 if (isSyslogFormat) {
204 final RFC5424Timestamp rfc5424Timestamp = new RFC5424Timestamp(rfc5424Frame.timestamp);
205 final JsonObject jsonEnvelope = eventMetadata.toJSONWithSyslogTimestamp(rfc5424Frame);
206 rowWriter.write(0, new EpochMicros(rfc5424Timestamp).asLong());
207 rowWriter.write(1, UTF8String.fromString(jsonEnvelope.toString()));
208 rowWriter.write(2, this.directory);
209 rowWriter.write(3, this.stream);
210 rowWriter.write(4, this.host);
211 rowWriter.write(5, new EventToSource().asUTF8StringFrom(rfc5424Frame));
212 rowWriter.write(6, this.id);
213 rowWriter.write(7, 0L);
214 rowWriter.write(8, new EventToOrigin().asUTF8StringFrom(rfc5424Frame));
215 }
216 else {
217 final JsonObject jsonEnvelope = eventMetadata.toJSONWithPathExtractedTimestamp();
218 rowWriter.write(0, new EpochMicros(new PathExtractedTimestamp(path)).asLong());
219 rowWriter.write(1, UTF8String.fromString(jsonEnvelope.toString()));
220 rowWriter.write(2, this.directory);
221 rowWriter.write(3, this.stream);
222 rowWriter.write(4, this.host);
223 rowWriter.write(5, "unknown-source".getBytes(StandardCharsets.UTF_8));
224 rowWriter.write(6, this.id);
225 rowWriter.write(7, 0L);
226 rowWriter.write(8, "unknown-origin".getBytes(StandardCharsets.UTF_8));
227 }
228 if (LOGGER.isDebugEnabled()) {
229 LOGGER.debug("Get Event, row=written");
230 }
231 return rowWriter.getRow();
232 }
