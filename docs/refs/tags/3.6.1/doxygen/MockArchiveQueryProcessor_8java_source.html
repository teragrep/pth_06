78 public MockArchiveQueryProcessor(String query) {

91 @Override
92 public Result<Record11<ULong, String, String, String, String, Date, String, String, Long, ULong, ULong>> processBetweenUnixEpochHours(
93 long startHour,
94 long endHour
95 ) {
96 LOGGER.info("MockArchiveQueryProcessor.range> " + startHour + " to " + endHour);
97 // start is inclusive and end is also inclusive
98
99 Result<Record11<ULong, String, String, String, String, Date, String, String, Long, ULong, ULong>> rv = generateResult(
100 null, null, null, null, null, null, null, null, null, null, null
101 );
102
103 final Timer.Context timerCtx = metricRegistry.timer("mockRowsTime").time();
104 for (long res : virtualDatabaseMap.keySet()) {
105 if (res > startHour && res <= endHour) {
106 rv.addAll(virtualDatabaseMap.get(res));
107 }
108 }
109 final long latencyNs = timerCtx.stop();
110
111 metricRegistry.histogram("mockRowTime").update((latencyNs / 1_000_000L) / rv.size());
112 SettableGauge<Long> count = metricRegistry.gauge("mockRowCount");
113 count.setValue((long) rv.size());
114 LOGGER.info("MockArchiveQueryProcessor.range> " + rv.formatCSV());
115 return rv;
116 }
117
118 @Override
119 public void commit(long offset) {
120 // end offset is committed
121 virtualDatabaseMap.keySet().removeIf(hour -> hour <= offset);
122 }
123
124 @Override
125 public Long incrementAndGetLatestOffset() {
126 Optional<Long> offset;
127 if (this.latestOffset == null) {
128 offset = virtualDatabaseMap
129 .keySet()
130 .stream()
131 .filter(x -> !x.equals(getInitialOffset()))
132 .sorted()
133 .findFirst();
134 }
135 else {
136 offset = virtualDatabaseMap.keySet().stream().filter(x -> x > this.latestOffset).sorted().findFirst();
137 }
138
139 // always return the last offset if later offset could not be found
140 this.latestOffset = offset.orElseGet(() -> this.latestOffset);
141 return this.latestOffset;
142 }
143
144 @Override
145 public Long mostRecentOffset() {
146 return latestOffset;
147 }
148
149 @Override
150 public CustomTaskMetric[] currentDatabaseMetrics() {

152 return new CustomTaskMetric[] {
153 new TaskMetric("ArchiveDatabaseRowCount", (long) metricRegistry.gauge("mockRowCount").getValue()),
154 new TaskMetric("ArchiveDatabaseRowMaxLatency", snapshot.getMax()),
155 new TaskMetric("ArchiveDatabaseRowAvgLatency", (long) snapshot.getMean()),
156 new TaskMetric("ArchiveDatabaseRowMinLatency", snapshot.getMin()),
157 };
158 }
