84 public List<Table<?>> tables() {
85 LOGGER.debug("Matching tables for condition <[{}]>, using ctx <{}>", condition, ctx);
86 TablesWithMatchingRegexAndNotEmpty tablesWithMatchingRegexAndNotEmpty = new TablesWithMatchingRegexAndNotEmpty(
87 ctx,
88 condition
89 );
90
91 Meta meta = ctx.meta();
92
93 if (LOGGER.isTraceEnabled()) {
94 LOGGER.trace("Found meta.getTables() <[{}]>", meta.getTables());
95 }
96
97 final List<Table<?>> tables = meta
98 .filterSchemas(s -> s.equals(BLOOMDB)) // select bloomdb
99 .filterTables(t -> !t.equals(BLOOMDB.FILTERTYPE)) // remove filtertype table
100 .filterTables(tablesWithMatchingRegexAndNotEmpty) // for each remaining table
101 .getTables();
102
103 LOGGER.debug("Table(s) with a pattern match <{}>", tables);
104
105 return tables;
106 }

118 @Override
119 public boolean test(final Table<?> t) {
120 // select id from each of the tables from the first row, then join to 'filtertype' via the filter_type_id, to access the pattern so that we know if the table is relevant for the query
121
122 SelectLimitPercentStep<Record1<ULong>> selectQuery = ctx
123 .select((Field<ULong>) t.field("id"))
124 .from(t)
125 .leftJoin(BLOOMDB.FILTERTYPE)// join filtertype to access patterns
126 .on(BLOOMDB.FILTERTYPE.ID.eq((Field<ULong>) t.field("filter_type_id")))
127 .where(condition.condition())// select tables that match the condition
128 .limit(1); // limit 1 since we are checking only if the table is not empty
129
130 LOGGER.trace("Testing a match for table <[{}]> using selectQuery <[{}]> ", t, selectQuery);
131
132 Result<Record1<ULong>> firstRowIdOnBloomCategoryTable = selectQuery.fetch();
133
134 LOGGER
135 .trace(
136 "Result for table <[{}]> firstRowIdOnBloomCategoryTable <[{}]>", t,
137 firstRowIdOnBloomCategoryTable
138 );
139
140 boolean isIncluded = firstRowIdOnBloomCategoryTable.isNotEmpty(); // select table if not empty
141
142 LOGGER.debug("Table <[{}]>, isIncluded <{}>", t, isIncluded);
143
144 return isIncluded;
145 }
