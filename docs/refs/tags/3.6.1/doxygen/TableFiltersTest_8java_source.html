166 Assertions.assertTrue(filter.mightContain("192.168.1.1"));

309 void fillTargetTable(int id) {

312 conn.prepareStatement("USE BLOOMDB").execute();

314 + "VALUES (?, (SELECT `id` FROM `filtertype` WHERE id=?), ?)";
