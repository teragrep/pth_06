222 private void writeFilter(String tableName, int filterId) {
223 Assertions.assertDoesNotThrow(() -> {
224 conn.prepareStatement("CREATE SCHEMA IF NOT EXISTS BLOOMDB").execute();
225 conn.prepareStatement("USE BLOOMDB").execute();
226 String sql = "INSERT INTO `" + tableName + "` (`partition_id`, `filter_type_id`, `filter`) "
227 + "VALUES (?, (SELECT `id` FROM `filtertype` WHERE id=?), ?)";
228 PreparedStatement stmt = conn.prepareStatement(sql);
229 BloomFilter filter = BloomFilter.create(1000, 0.01);
230 final ByteArrayOutputStream filterBAOS = new ByteArrayOutputStream();
231 Assertions.assertDoesNotThrow(() -> {
232 filter.writeTo(filterBAOS);
233 filterBAOS.close();
234 });
235 stmt.setInt(1, 1);
236 stmt.setInt(2, filterId);
237 stmt.setBytes(3, filterBAOS.toByteArray());
238 stmt.executeUpdate();
239 stmt.close();
240 });
241 }
