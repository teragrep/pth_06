102 logger.debug("NestedTopNQuery.getTableStatement called condition <{}> date <{}>", journaldbConditionArg, day);
103 SelectOnConditionStep<Record> selectOnConditionStep = DSL
104 .select(resultFields)
105 .from(GetArchivedObjectsFilterTable.FILTER_TABLE)
106 .innerJoin(JOURNALDB.LOGFILE.forceIndex(Indexes.LOGFILE_CIX_LOGFILE_LOGDATE_HOST_ID_LOGTAG_ID.getName())).on(JOURNALDB.LOGFILE.LOGTAG_ID.eq(GetArchivedObjectsFilterTable.tag_id)).and(JOURNALDB.LOGFILE.HOST_ID.eq(GetArchivedObjectsFilterTable.host_id));
107
108 if (streamDBClient.bloomEnabled()) {
109 // join all tables needed for the condition generated by walker
110 final Set<Table<?>> tables = streamDBClient.walker().conditionRequiredTables();
111 if (!tables.isEmpty()) {
112 for (final Table<?> table : tables) {
113 if (logger.isInfoEnabled()) {
114 logger.info("Left join pattern match table: <{}>", table.getName());
115 }
116 selectOnConditionStep = selectOnConditionStep
117 .leftJoin(table)
118 .on(JOURNALDB.LOGFILE.ID.eq((Field<ULong>) table.field("partition_id")));
119 }
120 }
121 }
122
123 logger.debug("NestedTopNQuery.getTableStatement exit");
124 return selectOnConditionStep
125 .where(JOURNALDB.LOGFILE.LOGDATE.eq(day).and(journaldbConditionArg))
126 .orderBy(logtimeForOrderBy, JOURNALDB.LOGFILE.ID.asc())
127 .asTable(innerTable);
