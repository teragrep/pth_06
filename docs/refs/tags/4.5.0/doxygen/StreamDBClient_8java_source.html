111 public StreamDBClient(Config config) throws SQLException {
112 this.isDebugEnabled = config.loggingConfig.isDebug();
113 this.LOGGER = new ConfiguredLogger(classLogger, isDebugEnabled);
114 this.isLogSQL = config.sqlConfig.isLog();
115 LOGGER.debug("StreamDBClient ctor called with config <[{}]>", config);
116 this.bloomEnabled = config.archiveConfig.bloomEnabled;
117 LOGGER.info("StreamDBClient bloom.enabled: " + this.bloomEnabled);
118
119 final String userName = config.archiveConfig.dbUsername;
120 final String password = config.archiveConfig.dbPassword;
121 final String url = config.archiveConfig.dbUrl;
122 final String journaldbName = config.archiveConfig.dbJournalDbName;
123 final String streamdbName = config.archiveConfig.dbStreamDbName;
124 final String bloomdbName = config.archiveConfig.bloomDbName;
125 final boolean isSQLThrowExceptionsNone = config.sqlConfig.isThrowExceptionsNone();
126 final boolean isSQLExecuteLogging = config.sqlConfig.isExecuteLogging();
127 final boolean withoutFilters = config.archiveConfig.withoutFilters;
128 final String withoutFiltersPattern = config.archiveConfig.withoutFiltersPattern;
129 // https://blog.jooq.org/how-i-incorrectly-fetched-jdbc-resultsets-again/
130 Settings settings = new Settings()
131 .withRenderMapping(new RenderMapping().withSchemata(new MappedSchema().withInput("streamdb").withOutput(streamdbName), new MappedSchema().withInput("journaldb").withOutput(journaldbName), new MappedSchema().withInput("bloomdb").withOutput(bloomdbName)));
132 if (isSQLThrowExceptionsNone) {
133 settings = settings.withThrowExceptions(ThrowExceptions.THROW_NONE);
134 LOGGER.warn("StreamDBClient SQL Exceptions set to THROW_NONE");
135 }
136
137 if (isSQLExecuteLogging) {
138 settings.withExecuteLogging(true);
139 }
140
141 System.getProperties().setProperty("org.jooq.no-logo", "true");
142 final Connection connection = DriverManager.getConnection(url, userName, password);
143 this.ctx = DSL.using(connection, SQLDialect.MYSQL, settings);
144 this.filterTable = new GetArchivedObjectsFilterTable(ctx, isDebugEnabled, isLogSQL);
145 this.nestedTopNQuery = new NestedTopNQuery(this, isDebugEnabled);
146 this.sliceTable = new SliceTable(ctx, isDebugEnabled, isLogSQL);
147
148 if (isSQLThrowExceptionsNone) {
149 // force sql mode to NO_ENGINE_SUBSTITUTION, STRICT mode
150
151 final String noEngineSubstitution = "SET sql_mode = 'NO_ENGINE_SUBSTITUTION';";
152 if (isLogSQL) {
153 LOGGER
154 .info(
155 "{SQL} StreamDBClient noEngineSubstitution <\n{}\n>", "noEngineSubstitution",
156 noEngineSubstitution
157 );
158 }
159 ctx.execute(noEngineSubstitution);
160 }
161
162 // -- TODO use dslContext.batch for all initial operations
163 final FilterlessSearch filterlessSearch;
164 if (withoutFilters) {
165 filterlessSearch = new FilterlessSearchImpl(ctx, withoutFiltersPattern);
166 }
167 else {
168 filterlessSearch = new FilterlessSearchStub();
169 }
170
171 this.walker = new ConditionWalker(ctx, bloomEnabled, filterlessSearch);
172 Condition streamdbCondition;
173
174 try {
175 // Construct both streamDB and journalDB query conditions
176 streamdbCondition = walker.fromString(config.query, true);
177 this.journaldbCondition = walker.fromString(config.query, false);
178 }
179 catch (ParserConfigurationException | IOException | SAXException e) {
180 throw new IllegalArgumentException(e);
181 }
182
183 filterTable.create(streamdbCondition); // TEMPTABLE
184 sliceTable.create();
185
186 // by default no cutoff
187 includeBeforeEpoch = config.archiveConfig.archiveIncludeBeforeEpoch;
188
189 this.metricRegistry = new MetricRegistry();
190 LOGGER.debug("StreamDBClient ctor exit");
191 }

204 public int pullToSliceTable(Date day) {
205 LOGGER.debug("StreamDBClient.pullToSliceTable called for date <{}>", day);
206
207 SelectOnConditionStep<Record10<ULong, String, String, String, Date, String, String, Long, ULong, ULong>> select = ctx
208 .select(
209 JOURNALDB.LOGFILE.ID, nestedTopNQuery.directory(), nestedTopNQuery.stream(),
210 JOURNALDB.HOST.NAME, JOURNALDB.LOGFILE.LOGDATE, JOURNALDB.BUCKET.NAME, JOURNALDB.LOGFILE.PATH,
211 nestedTopNQuery.logtime(), JOURNALDB.LOGFILE.FILE_SIZE, JOURNALDB.LOGFILE.UNCOMPRESSED_FILE_SIZE
212 )
213 .from(nestedTopNQuery.getTableStatement(journaldbCondition, day))
214 .join(JOURNALDB.LOGFILE)
215 .on(JOURNALDB.LOGFILE.ID.eq(nestedTopNQuery.id()))
216 .join(JOURNALDB.BUCKET)
217 .on(JOURNALDB.BUCKET.ID.eq(JOURNALDB.LOGFILE.BUCKET_ID))
218 .join(JOURNALDB.HOST)
219 .on(JOURNALDB.HOST.ID.eq(JOURNALDB.LOGFILE.HOST_ID))
220 .join(JOURNALDB.LOGTAG)
221 .on(JOURNALDB.LOGTAG.ID.eq(JOURNALDB.LOGFILE.LOGTAG_ID));
222
223 final Timer.Context timerCtx = metricRegistry.timer("ArchiveDatabaseLatency").time();
224 final int rows;
225
226 try (final InsertOnDuplicateStep<Record> selectStep = ctx.insertInto(SliceTable.SLICE_TABLE).select(select)) {
227 if (isLogSQL) {
228 LOGGER
229 .info(
230 "{SQL} StreamDBClient.pullToSliceTable selectStep <\n{}\n>",
231 selectStep.getSQL(ParamType.INLINED)
232 );
233 }
234 rows = selectStep.execute();
235 }
236
237 final long latencyNs = timerCtx.stop();
238
239 if (rows != 0) {
240 metricRegistry.histogram("ArchiveDatabaseLatencyPerRow").update(latencyNs / rows);
241 }
242
243 LOGGER.info("StreamDBClient.pullToSliceTable" + ": took (" + "<{}> ms)", (latencyNs / 1_000_000d));
244
245 metricRegistry.counter("ArchiveDatabaseRowCount").inc(rows);
246
247 LOGGER.debug("StreamDBClient.pullToSliceTable returns <{}> rows", rows);
248 return rows;
249
250 }

252 WeightedOffset getNextHourAndSizeFromSliceTable(long previousHour) {
253 LOGGER.debug("StreamDBClient.getNextHourAndSizeFromSliceTable called with previousHour <{}>", previousHour);
254
255 final Result<Record2<Long, BigDecimal>> hourAndFilesizeRecord = ctx
256 .select(SliceTable.logtime, DSL.sum(SliceTable.filesize))
257 .from(SliceTable.SLICE_TABLE)
258 .where(SliceTable.logtime.greaterThan(previousHour).and(SliceTable.logtime.lessThan(includeBeforeEpoch))).groupBy(SliceTable.logtime).orderBy(SliceTable.logtime.asc()).limit(1).fetch();
259 final WeightedOffset weightedOffset;
260 if (hourAndFilesizeRecord.isEmpty()) {
261 weightedOffset = new WeightedOffset();
262 }
263 else {
264 final long offset = hourAndFilesizeRecord.get(0).get(0, Long.class);
265 final long fileSize = hourAndFilesizeRecord.get(0).get(1, BigDecimal.class).longValue();
266
267 weightedOffset = new WeightedOffset(offset, fileSize);
268 }
269
270 LOGGER.debug("StreamDBClient.getNextHourAndSizeFromSliceTable returns weightedOffset <{}>", weightedOffset);
271 return weightedOffset;
