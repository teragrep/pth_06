138 final StreamingQuery streamingQuery = Assertions
139 .assertDoesNotThrow(() -> df.writeStream().outputMode(OutputMode.Append()).format("memory").trigger(Trigger.ProcessingTime(0)).queryName("MockArchiveQuery").option("checkpointLocation", "/tmp/checkpoint/" + UUID.randomUUID()).option("spark.cleaner.referenceTracking.cleanCheckpoints", "true").start());

149
150 @Override
151 public void onQueryStarted(final QueryStartedEvent event) {
152 // no-op

154
155 @Override
156 public void onQueryProgress(final QueryProgressEvent event) {
157 if (id.equals(event.progress().id())) {
158 latestOffset[0] = event.progress().sources()[0].latestOffset();
159 }
160 }
161
162 @Override
163 public void onQueryIdle(final QueryIdleEvent event) {
164 if (id.equals(event.id())) {
165 triggered[0]++;
166 Assertions.assertDoesNotThrow(streamingQuery::stop);
167 }
168 }
169
170 @Override
171 public void onQueryTerminated(final QueryTerminatedEvent event) {
172 // no-op
173 }
174 });
