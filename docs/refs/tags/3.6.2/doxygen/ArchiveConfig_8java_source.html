76 public ArchiveConfig(Map<String, String> opts) {
77 s3EndPoint = getOrThrow(opts, "S3endPoint");
78 s3Identity = getOrThrow(opts, "S3identity");
79 s3Credential = getOrThrow(opts, "S3credential");
80 dbUsername = getOrThrow(opts, "DBusername");
81 dbPassword = getOrThrow(opts, "DBpassword");
82 dbUrl = getOrThrow(opts, "DBurl");
83 bloomEnabled = opts.getOrDefault("bloom.enabled", "false").equalsIgnoreCase("true");
84 withoutFilters = opts.getOrDefault("bloom.withoutFilters", "false").equalsIgnoreCase("true");
85 withoutFiltersPattern = opts.getOrDefault("bloom.withoutFiltersPattern", "");
86 bloomDbName = opts.getOrDefault("DBbloomdbname", "bloomdb");
87
88 dbJournalDbName = opts.getOrDefault("DBjournaldbname", "journaldb");
89 dbStreamDbName = opts.getOrDefault("DBstreamdbname", "streamdb");
90
91 // hide exceptions
92 hideDatabaseExceptions = opts.getOrDefault("hideDatabaseExceptions", "false").equalsIgnoreCase("true");
93
94 // skip not rfc5424 parseable files
95 skipNonRFC5424Files = opts.getOrDefault("skipNonRFC5424Files", "false").equalsIgnoreCase("true");
96
97 archiveIncludeBeforeEpoch = Long
