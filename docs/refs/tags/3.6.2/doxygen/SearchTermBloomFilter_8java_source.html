82 public byte[] bytes() {
83 LOGGER.debug("Create filter from Record with values: expected <{}>, fpp <{}>", expected, fpp);
84 if (stringTokens.isEmpty()) {
85 throw new IllegalStateException("Tried to create a filter without any items");
86 }
87 final BloomFilter filter = BloomFilter.create(expected, fpp);
88 for (final String token : stringTokens) {
89 filter.put(token.toLowerCase());
90 }
91 try (final ByteArrayOutputStream filterBAOS = new ByteArrayOutputStream()) {
92 filter.writeTo(filterBAOS);
93 return filterBAOS.toByteArray();
94 }
95 catch (final IOException e) {
96 throw new UncheckedIOException("Error writing filter bytes: ", e);
97 }
98 }
