67 public Condition condition() {

73 condition = StreamDBClient.GetArchivedObjectsFilterTable.host.like(value.replace('*', '%').toLowerCase());

75 if ("NOT_EQUALS".equalsIgnoreCase(operation)) {
76 condition = condition.not();
