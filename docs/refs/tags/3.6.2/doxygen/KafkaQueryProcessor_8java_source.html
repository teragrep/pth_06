85 public KafkaQueryProcessor(Config config) {
86 this.persistedEndOffsetMap = new HashMap<>();
87 this.continuousProcessing = config.kafkaConfig.kafkaContinuousProcessing;
88 Map<String, Object> executorKafkaProperties = config.kafkaConfig.driverOpts;
89 this.kafkaConsumer = new KafkaConsumer<>(executorKafkaProperties);
90
91 String topicsRegexString = null;
92 if (config.query != null) {
93 try {
94 KafkaWalker parser = new KafkaWalker();
95 topicsRegexString = parser.fromString(config.query);
96 }
97 catch (ParserConfigurationException | IOException | SAXException ex) {
98 ex.printStackTrace();
99 throw new RuntimeException(
100 "KafkaQueryProcessor problems when construction Query conditions query:" + config.query
101 + " exception:" + ex
102 );
103 }
104 }
105
106 if (topicsRegexString == null) {
107 topicsRegexString = "^.*$"; // all topics if none given
108 }
109
110 Pattern topicsRegex = Pattern.compile(topicsRegexString);
111 LOGGER.info("Subscribing to: " + topicsRegexString);
112 kafkaConsumer.subscribe(topicsRegex);
113
114 Set<TopicPartition> localTopicPartitionSet = Collections.emptySet();
115 while (localTopicPartitionSet.isEmpty()) {
116 try {
117 kafkaConsumer.poll(Duration.ofSeconds(60));
118 }
119 catch (TopicAuthorizationException topicAuthorizationException) {
120 handleTopicAuthorizationException(topicsRegexString, topicAuthorizationException);
121 }
122 localTopicPartitionSet = kafkaConsumer.assignment();
123
124 if (localTopicPartitionSet.isEmpty()) {
125 LOGGER.warn("kafkaConsumer.assignment() returned no topics, verifying subscription and retrying");
126
127 // check if connectivity issue or if illegal pattern
128 Map<String, List<PartitionInfo>> listTopics = kafkaConsumer.listTopics(Duration.ofSeconds(60));
129 boolean hasTopicMatch = false;
130 for (Map.Entry<String, List<PartitionInfo>> entry : listTopics.entrySet()) {
131 Matcher matcher = topicsRegex.matcher(entry.getKey());
132 if (matcher.matches()) {
133 hasTopicMatch = true;
134 }
135 }
136
137 if (!hasTopicMatch && !listTopics.isEmpty()) {
138 //throw new IllegalStateException("subscribed to non-existing topic");
139 // passthrough
140 localTopicPartitionSet = Collections.emptySet();
141 LOGGER.warn("subscribed to topics that do not exist! Kafka data will not be available.");
142 break;
143 }
144
145 }
146 }
147 topicPartitionSet = localTopicPartitionSet;
148 LOGGER.debug("KafkaQueryProcessor");
149 }

151 private void handleTopicAuthorizationException(
152 String topicsRegexString,
153 TopicAuthorizationException topicAuthorizationException
154 ) {
155 // wildcard queries may match topics with no authorization
156 Set<String> unauthorizedTopics = topicAuthorizationException.unauthorizedTopics();
157
158 // use negative lookahead to cancel unauthorizedTopics:
159 // by prepending it to authorized topics
160 // i.e (?!^not_a|not_b|not_c$)^(yes_x|(yes_y|yes_z))$
161 StringBuilder stringBuilder = new StringBuilder();
162 stringBuilder.append("(?!");
163
164 int topicsProcessed = 0;
165 for (String topic : unauthorizedTopics) {
166 stringBuilder.append("^");
167 stringBuilder.append(topic);
168 stringBuilder.append("$");
169 topicsProcessed++;
170
171 if (topicsProcessed < unauthorizedTopics.size()) {
172 stringBuilder.append("|");
173 }
174 }
175 stringBuilder.append(")");
176
177 stringBuilder.append(topicsRegexString);
178
179 String negatedTopicRegexString = stringBuilder.toString();
180
181 Pattern negatedTopicRegexPattern = Pattern.compile(negatedTopicRegexString);
182
183 LOGGER
184 .warn(
185 "Re-subscribing: <[" + negatedTopicRegexPattern + "]>" + " after TopicAuthorizationException: "
186 + topicAuthorizationException
187 );
188 kafkaConsumer.subscribe(negatedTopicRegexPattern);
189
190 // it seems that kafka may not report all unauthorized topics in a
191 // one go, therefore recursion is needed
192 try {
193 kafkaConsumer.poll(Duration.ofSeconds(60));
194 }
195 catch (TopicAuthorizationException authorizationException) {
196 handleTopicAuthorizationException(negatedTopicRegexString, authorizationException);
197 }
198 }

209 @Override
210 public Map<TopicPartition, Long> getInitialEndOffsets() {
211 if (persistedEndOffsetMap.isEmpty() || continuousProcessing) {
212
213 Map<TopicPartition, Long> topicPartitionEndOffsetMap = new HashMap<>();
214 Map<TopicPartition, Long> endOffsets = kafkaConsumer.endOffsets(topicPartitionSet, Duration.ofSeconds(60));
215
216 for (TopicPartition topicPartition : topicPartitionSet) {
217 long partitionEnd = endOffsets.get(topicPartition);
218 topicPartitionEndOffsetMap.put(topicPartition, partitionEnd);
219
220 }
221 persistedEndOffsetMap.putAll(topicPartitionEndOffsetMap);
222 return topicPartitionEndOffsetMap;
223 }
