110 public T traverse(TreeWalker walker, String op) throws IllegalStateException {
111 Node parend = walker.getCurrentNode();
112 Element current = ((Element) parend);
113 T rv = null;
114 LOGGER.debug(" traverse incoming:" + current.getTagName());
115
116 try {
117 if (
118 current.getTagName().equalsIgnoreCase("AND") || current.getTagName().equalsIgnoreCase("OR")
119 || current.getTagName().equalsIgnoreCase("NOT")
120 ) {
121 op = current.getTagName();
122 }
123
124 if (parend.hasChildNodes()) {
125 NodeList children = parend.getChildNodes();
126 int count = children.getLength();
127 if (count < 1 || count > 2) {
128 throw new IllegalStateException("Error, wrong number of children:" + count + " op:" + op);
129 }
130 // get left and right
131 Node left = walker.firstChild();
132 T lft;
133 switch (count) {
134 case 1: {
135 LOGGER.debug(" 1 child incoming:" + current + " left=" + left + " op:" + op);
136
137 //rv = emitLogicalOperation(op, lft, null);
138 walker.setCurrentNode(left);
139 lft = traverse(walker, op);
140 LOGGER.debug("--Traverse 1 child op:" + op + " lft=" + lft);
141 rv = lft;
142 break;
143 }
144 case 2: {
145 lft = traverse(walker, op);
146 T rht = null;
147 Node right = walker.nextSibling();
148 walker.setCurrentNode(left);
149 // System.out.println("traverse right:"+right);
150 if (right != null) {
151 walker.setCurrentNode(right);
152 rht = traverse(walker, op);
153 }
154 if (lft != null && rht != null) {
155 if (op == null) {
156 throw new IllegalStateException("Parse error, unbalanced elements. " + lft.toString());
157 }
158 rv = emitLogicalOperation(op, lft, rht);
159 }
160 else if (lft == null) {
161 rv = rht;
162 }
163 else if (rht == null) {
164 rv = lft;
165 }
166 break;
167 }
168 default:
169 }
170 }
171 else {
172 // leaf
173 if (op != null && "NOT".equals(op)) {
174 LOGGER.debug("Emit Unary operation op:" + op + " l:" + current);
175 rv = emitUnaryOperation(op, current);
176 }
177 else {
178 LOGGER.debug("EmitElem");
179 rv = emitElem(current);
180 }
181 }
182 walker.setCurrentNode(parend);
183 //if(rv != null)
184 // System.out.println("XmlWalker.traverse type:"+rv.getClass().getName() +" returns:"+((T)rv).toString());
185 return (T) rv;
186 }
187 catch (IllegalStateException e) {
188 LOGGER.error(e.toString());
189 }
190 return null;
191 }

223 public static String escapeSpecialCharacters(String input) {
224 final List<String> specialCharacters = Arrays.asList("\\", "*", "+", "?", "%");
225 if ("*".equals(input)) {
226 return "%";
227 }
228 else {
229 return Arrays.stream(input.split("")).map((c) -> {
230 if (specialCharacters.contains(c)) {
231 return "\\" + c;
232 }
233 else {
234 return c;
235 }
236 }).collect(Collectors.joining());
237 }
