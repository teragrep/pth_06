73 public Batch(Config config, ArchiveQuery aq, KafkaQuery kq) {
74 this.config = config;
75 this.runQueueArray = new LinkedList<>();
76
77 for (int i = 0; i < config.batchConfig.numPartitions; i++) {
78 this.runQueueArray.add(new BatchTaskQueue());
79 }
80
81 this.archiveQuery = aq;
82 this.kafkaQuery = kq;
83 }

85 public Batch processRange(Offset start, Offset end) {
86 LOGGER.debug("processRange");
87
88 BatchSliceCollection slice = null;
89 if (config.isArchiveEnabled) {
90 slice = new ArchiveBatchSliceCollection(this.archiveQuery).processRange(start, end);
91 }
92
93 if (config.isKafkaEnabled) {
94 if (slice == null) {
95 slice = new KafkaBatchSliceCollection(this.kafkaQuery).processRange(start, end);
96 }
97 else {
98 slice.addAll(new KafkaBatchSliceCollection(this.kafkaQuery).processRange(start, end));
99 }
100 }
101 if (slice != null && !slice.isEmpty()) {
102 this.addSlice(slice);
103 }
104
105 return this.getBatch();
106 }

108 public void addSlice(LinkedList<BatchSlice> sliceCollection) {
109 numberOfBatches++;
110
111 while (!sliceCollection.isEmpty()) {
112 BatchSlice longestObject = null;
113
114 // find the longest object
115 for (BatchSlice objectMetadata : sliceCollection) {
116 if (longestObject == null) {
117 longestObject = objectMetadata;
118 }
119 else {
120 if (longestObject.getSize() > objectMetadata.getSize()) {
121 longestObject = objectMetadata;
122 }
123 }
124 }
125
126 // object found
127 if (longestObject != null) {
128 sliceCollection.remove(longestObject);
129
130 // find shortest queue
131 BatchTaskQueue shortestQueue = null;
132 for (BatchTaskQueue btq : runQueueArray) {
133 if (shortestQueue == null) {
134 shortestQueue = btq;
135 }
136 else {
137 if (shortestQueue.getQueueTime() > btq.getQueueTime()) {
138 shortestQueue = btq;
139 }
140 }
141 }
142 if (shortestQueue != null) {
143 shortestQueue.add(longestObject);
144 }
145 else {
146 throw new RuntimeException("shortestQueue was null");
147 }
148
149 }
150 }
151 }
152
153 public Batch getBatch() {
154 this.clear(); // clear on getBatch?
155 for (BatchTaskQueue btq : runQueueArray) {
156 this.add(btq.getQueue());
157 }
158 LOGGER.debug("getBatch: " + this);
159 return this;
160 }
