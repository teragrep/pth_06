78 public Condition condition() {
79 final String tag = element.tag();
80 final String value = element.value();
81 final String operation = element.operation();
82 Condition condition = DSL.noCondition();
83 switch (tag.toLowerCase()) {
84 case "index":
85 QueryCondition index = new IndexCondition(value, operation, config.streamQuery());
86 condition = index.condition();
87 break;
88 case "sourcetype":
89 QueryCondition sourceType = new SourceTypeCondition(value, operation, config.streamQuery());
90 condition = sourceType.condition();
91 break;
92 case "host":
93 QueryCondition host = new HostCondition(value, operation, config.streamQuery());
94 condition = host.condition();
95 break;
96 default:
97 LOGGER.debug("Element tag was not index, sourcetype or host: <{}>", tag);
98 }
99 if (!config.streamQuery()) {
100 // Handle also time qualifiers
101 if ("earliest".equalsIgnoreCase(tag) || "index_earliest".equalsIgnoreCase(tag)) {
102 QueryCondition earliest = new EarliestCondition(value);
103 condition = earliest.condition();
104 }
105 if ("latest".equalsIgnoreCase(tag) || "index_latest".equalsIgnoreCase(tag)) {
106 QueryCondition latest = new LatestCondition(value);
107 condition = latest.condition();
108 }
109 // value search
110 if ("indexstatement".equalsIgnoreCase(tag) && "EQUALS".equals(operation) && config.bloomEnabled()) {
111 QueryCondition indexStatement = new IndexStatementCondition(value, config, condition);
112 condition = indexStatement.condition();
113 }
114 }
115 // bloom search can return the condition unmodified
116 if (condition.equals(DSL.noCondition()) && !isBloomSearchCondition()) {
117 throw new IllegalStateException("Unsupported Element tag " + tag);
118 }
119 LOGGER.debug("Query condition: <{}>", condition);
120 return condition;
121 }
122
123 public boolean isBloomSearchCondition() {
124 final String tag = element.tag();
125 final String operation = element.operation();
126 return "indexstatement".equalsIgnoreCase(tag) && "EQUALS".equals(operation) && !config.streamQuery()
127 && config.bloomEnabled();
128 }
129
131 public Set<Table<?>> requiredTables() {
132 final String value = element.value();
133 return new IndexStatementCondition(value, config).requiredTables();
134 }
