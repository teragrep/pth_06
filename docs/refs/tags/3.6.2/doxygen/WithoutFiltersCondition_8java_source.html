62public final class WithoutFiltersCondition implements QueryCondition, BloomQueryCondition {

76 private WithoutFiltersCondition(
77 final DSLContext ctx,
78 final String withoutFiltersPattern,
79 final Set<Table<?>> tables
80 ) {
81 this.ctx = ctx;
82 this.withoutFiltersPattern = withoutFiltersPattern;
83 this.tables = tables;
84 }
85
86 @Override
87 public Condition condition() {
88 if (tables.isEmpty()) {
89
90 final QueryCondition tableFilteringCondition = new StringEqualsCondition(
91 withoutFiltersPattern,
92 BLOOMDB.FILTERTYPE.PATTERN
93 );
94
95 final DatabaseTables conditionMatchingTables = new ConditionMatchBloomDBTables(
96 ctx,
97 tableFilteringCondition
98 );
99
100 tables.addAll(conditionMatchingTables.tables());
101 }
102
103 Condition combinedNullFilterCondition = DSL.noCondition();
104 for (final Table<?> table : tables) {
105 final Condition nullFilterCondition = table.field("filter").isNull();
106 combinedNullFilterCondition = combinedNullFilterCondition.and(nullFilterCondition);
107 }
108
109 return combinedNullFilterCondition;
110 }
111
112 @Override
113 public boolean isBloomSearchCondition() {
114 return false;
115 }
116
117 @Override
118 public Set<Table<?>> requiredTables() {
119 if (tables.isEmpty()) {
120 condition();
121 }
122 return tables;
123 }

140 }
