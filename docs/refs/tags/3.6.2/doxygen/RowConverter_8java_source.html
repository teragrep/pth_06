163 public void open() throws IOException {
164 // Open s3-file
165
166 String bucketName = this.bucket;
167 String keyName = this.path;
168
169 String logName = bucketName + "/" + keyName;
170 S3Object s3object;
171 try {
172 LOGGER.debug("Attempting to open file: " + logName);
173 s3object = s3client.getObject(bucketName, keyName);
174
175 if (LOGGER.isDebugEnabled()) {
176 LOGGER
177 .debug(
178 "Open S3 stream bucket:" + bucketName + " keyname=" + keyName + " Metadata length:"
179 + s3object.getObjectMetadata().getContentLength()
180 );
181 }
182 this.inputStream = new BufferedInputStream(s3object.getObjectContent(), 8 * 1024 * 1024);
183 GZIPInputStream gz = new GZIPInputStream(inputStream);
184 rfc5424Frame.load(gz);
185
186 // Set up result
187 LOGGER.trace("S3FileHandler.open() Initialized result set with element lists");
188
189 LOGGER.info("S3FileHandler.open() Initialized parser for: " + logName);
190 }
191 catch (AmazonServiceException amazonServiceException) {
192
193 if (403 == amazonServiceException.getStatusCode()) {
194 LOGGER
195 .error(
196 "Skipping file " + logName + " due to errorCode:"
197 + amazonServiceException.getStatusCode()
198 );
199 }
200 else {
201 throw amazonServiceException;
202 }
203 }

228 LOGGER.error("ParseException at object: <[{}]>/<[{}]>", bucket, path);
229 if (skipNonRFC5424Files) {
230 rv = false;
231 }
232 else {
233 throw parseException;
234 }
235 }

239 static long rfc3339ToEpoch(ZonedDateTime zonedDateTime) {
240 final Instant instant = zonedDateTime.toInstant();
241
242 final long MICROS_PER_SECOND = 1000L * 1000L;
243 final long NANOS_PER_MICROS = 1000L;
244 final long sec = Math.multiplyExact(instant.getEpochSecond(), MICROS_PER_SECOND);
245
246 return Math.addExact(sec, instant.getNano() / NANOS_PER_MICROS);
247 }

249 public InternalRow get() {
250 //System.out.println("RowConverter.get>");
251 if (LOGGER.isDebugEnabled()) {
252 LOGGER
253 .debug(
254 "RowConverter.get> Partition (" + this.id + "):" + bucket + "/" + path + " Get("
255 + currentOffset + ")"
256 );
257 }
258
259 if (LOGGER.isDebugEnabled()) {
260 LOGGER.debug("Parsersyslog event:" + rfc5424Frame.toString());
261 }
262
263 final long epochMicros = rfc3339ToEpoch(new RFC5424Timestamp(rfc5424Frame.timestamp).toZonedDateTime());
264
265 // input
266 final byte[] source = eventToSource();
267
268 // origin
269 final byte[] origin = eventToOrigin();
270
271 if (LOGGER.isDebugEnabled()) {
272 LOGGER
273 .trace(
274 "PARSED epochMicros: " + epochMicros + " message: "
275 + new String(rfc5424Frame.msg.toBytes(), StandardCharsets.UTF_8)
276 );
277 }
278
279 rowWriter.reset();
280 rowWriter.zeroOutNullBytes();
281 rowWriter.write(0, epochMicros);
282 rowWriter.write(1, UTF8String.fromBytes(rfc5424Frame.msg.toBytes()));
283 rowWriter.write(2, this.directory);
284 rowWriter.write(3, this.stream);
285 rowWriter.write(4, this.host);
286 rowWriter.write(5, UTF8String.fromBytes(source));
287 rowWriter.write(6, this.id);
288 rowWriter.write(7, currentOffset);
289 rowWriter.write(8, UTF8String.fromBytes(origin));
290
291 if (LOGGER.isDebugEnabled()) {
292 LOGGER.debug("Get Event, row=written");
293 }
294
295 auditPlugin
296 .audit(
297 epochMicros, rfc5424Frame.msg.toBytes(), this.directory.getBytes(), this.stream.getBytes(),
298 this.host.getBytes(), source, this.id.toString(), currentOffset
299 );
300 return rowWriter.getRow();
301
302 }

304 private byte[] eventToOrigin() {
305 byte[] origin;

311 origin = new byte[] {};

316 private byte[] eventToSource() {
317 //input is produced from SD element event_node_source@48577 by
318 // concatenating "source_module:hostname:source". in case
319 //if event_node_source@48577 is not available use event_node_relay@48577.
320 //If neither are present, use null value.
321
322 sourceConcatenationBuffer.clear();
323
324 Fragment sourceModuleFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceSourceModule);
325 if (sourceModuleFragment.isStub) {
326 sourceModuleFragment = rfc5424Frame.structuredData.getValue(eventNodeRelaySourceModule);
327 }
328
329 byte[] source_module;
330 if (!sourceModuleFragment.isStub) {
331 source_module = sourceModuleFragment.toBytes();
332 }
333 else {
334 source_module = new byte[] {};
335 }
336
337 Fragment sourceHostnameFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceHostname);
338 if (sourceHostnameFragment.isStub) {
339 sourceHostnameFragment = rfc5424Frame.structuredData.getValue(eventNodeRelayHostname);
340 }
341
342 byte[] source_hostname;
343 if (!sourceHostnameFragment.isStub) {
344 source_hostname = sourceHostnameFragment.toBytes();
345 }
346 else {
347 source_hostname = new byte[] {};
348 }
349
350 Fragment sourceSourceFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceSource);
351 if (sourceHostnameFragment.isStub) {
352 sourceSourceFragment = rfc5424Frame.structuredData.getValue(eventNodeRelaySource);
353 }
354
355 byte[] source_source;
356 if (!sourceSourceFragment.isStub) {
357 source_source = sourceSourceFragment.toBytes();
358 }
359 else {
360 source_source = new byte[] {};
361 }
362
363 // source_module:hostname:source"
364 sourceConcatenationBuffer.put(source_module);
365 sourceConcatenationBuffer.put((byte) ':');
366 sourceConcatenationBuffer.put(source_hostname);
367 sourceConcatenationBuffer.put((byte) ':');
368 sourceConcatenationBuffer.put(source_source);
369
370 sourceConcatenationBuffer.flip();
371 byte[] input = new byte[sourceConcatenationBuffer.remaining()];
372 sourceConcatenationBuffer.get(input);
373
374 return input;
375 }
