104 public Condition fromString(String inXml, boolean streamQueryArg)
105 throws IllegalStateException, ParserConfigurationException, IOException, SAXException {
106 this.streamQuery = streamQueryArg;
107 final Condition walkerCondition = fromString(inXml);
108 final Condition returnedCondition;
109 if (!filterlessSearch.isStub()) {
110 final WithoutFiltersCondition withoutFiltersQueryCondition = filterlessSearch.condition();
111 conditionRequiredTables().addAll(withoutFiltersQueryCondition.requiredTables());
112 final Condition withoutFiltersCondition = withoutFiltersQueryCondition.condition();
113 LOGGER.debug("Without filters option enabled with condition <{}>", withoutFiltersCondition);
114 returnedCondition = withoutFiltersCondition.and(walkerCondition);
115 }
116 else {
117 returnedCondition = walkerCondition;
118 }
119
120 return returnedCondition;
121 }

128 @Override
129 public Condition emitLogicalOperation(String op, Object l, Object r) throws IllegalStateException {
130 Condition rv;
131 Condition left = (Condition) l;
132 Condition right = (Condition) r;
133
134 if (op == null) {
135 throw new IllegalStateException("Parse error, unbalanced elements. " + left.toString());
136 }
137 if ("AND".equalsIgnoreCase(op)) {
138 rv = left.and(right);
139 }
140 else if ("OR".equalsIgnoreCase(op)) {
141 rv = left.or(right);
142 }
143 else if ("NOT".equalsIgnoreCase(op)) {
144 rv = left.not();
145 }
146 else {
147 throw new IllegalStateException(
148 "Parse error, unssorted logical operation. op:" + op + " expression:" + left.toString()
149 );
150 }
151 return rv;
152 }
153
154 @Override
155 public Condition emitUnaryOperation(String op, Element current) throws IllegalStateException {
156
157 Condition rv = emitElem(current);
158
159 LOGGER.info("ConditionWalker.emitUnaryOperation incoming op:" + op + " element:" + current);
160
161 if (op == null) {
162 throw new IllegalStateException("Parse error, op was null");
163 }
164 if (rv != null) {
165 if ("NOT".equalsIgnoreCase(op)) {
166 rv = rv.not();
167 }
168 else {
169 throw new IllegalStateException(
170 "Parse error, unsupported logical operation. op:" + op + " expression:" + rv.toString()
171 );
172 }
173 }
174 return rv;
175 }
176
177 Condition emitElem(final Element current) {
178 final ValidElement element = new ValidElement(current);
179 final ConditionConfig conditionConfig = new ConditionConfig(ctx, streamQuery, bloomEnabled, bloomTermId);
180 final ElementCondition elementCondition = new ElementCondition(element, conditionConfig);
181 if (elementCondition.isBloomSearchCondition() && !filterlessSearch.isStub()) {
182 throw new RuntimeException("Search terms are not allowed when <bloom.withoutFilters> option is enabled");
183 }
184 if (elementCondition.isBloomSearchCondition()) {
185 final Set<Table<?>> conditionRequiredTables = elementCondition.requiredTables();
186 // add tables condition found to walker pattern match tables
187 conditionRequiredTables().addAll(conditionRequiredTables);
188 bloomTermId++;
189 }
190 return elementCondition.condition();
191 }
