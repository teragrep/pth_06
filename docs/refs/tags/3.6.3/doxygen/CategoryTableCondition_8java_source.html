66 public CategoryTableCondition(Table<?> comparedTo, long bloomTermId) {
67 this(
68 comparedTo,
69 DSL.field("term_id", BIGINTUNSIGNED.nullable(false)).eq(ULong.valueOf(bloomTermId)),
70 DSL.field("type_id", BIGINTUNSIGNED.nullable(false)).eq((Field<ULong>) comparedTo.field("filter_type_id")), DSL.table(DSL.name(("term_" + bloomTermId + "_" + comparedTo.getName())))
71 );
72 }

74 public CategoryTableCondition(
75 Table<?> comparedTo,

86 public Condition condition() {
87 final Field<byte[]> filterField = DSL.field(DSL.name(categoryTable.getName(), "filter"), byte[].class);
88 // select filter with the correct bloom term id and filter type id from the category table
89 final SelectConditionStep<Record1<byte[]>> selectFilterStep = DSL
90 .select(filterField)
91 .from(categoryTable)
92 .where(bloomTermCondition)
93 .and(typeIdCondition);
94 // function 'bloommatch' compares category table filter byte[] against bloom filter byte[]
95 final Condition filterFieldCondition = DSL
96 .function("bloommatch", Boolean.class, selectFilterStep.asField(), comparedTo.field("filter"))
97 .eq(true);
98 // null check allows SQL to optimize query
99 final Condition notNullCondition = comparedTo.field("filter").isNotNull();
100 return filterFieldCondition.and(notNullCondition);
101 }
