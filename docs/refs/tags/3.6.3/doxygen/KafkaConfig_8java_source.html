69 public KafkaConfig(Map<String, String> opts) {
70 consumerGroupId = UUID.randomUUID().toString();
71 driverOpts = parseKafkaDriverOpts(opts);
72 executorOpts = parseKafkaExecutorOpts(opts);
73 executorConfig = parseKafkaExecutorConfig(opts);

93 private Map<String, Object> parseKafkaCommonOpts(Map<String, String> opts) {
94 // options common for driver and executor
95 Map<String, Object> rv = new HashMap<>();
96 String bootstrapServers = opts.get("kafka.bootstrap.servers");
97
98 if (bootstrapServers == null) {
99 throw new IllegalArgumentException("missing kafka.bootstrap.servers");
100 }
101 rv.put(ConsumerConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
102
103 String saslMechanism = opts.get("kafka.sasl.mechanism");
104 if (saslMechanism == null) {
105 throw new IllegalArgumentException("missing kafka.sasl.mechanism");
106 }
107 rv.put(SaslConfigs.SASL_MECHANISM, saslMechanism);
108
109 String securityProtocol = opts.get("kafka.security.protocol");
110 if (securityProtocol == null) {
111 throw new IllegalArgumentException("missing kafka.security.protocol");
112 }
113 rv.put(CommonClientConfigs.SECURITY_PROTOCOL_CONFIG, securityProtocol);
114
115 String jaasConfig = opts.get("kafka.sasl.jaas.config");
116 if (jaasConfig == null) {
117 throw new IllegalArgumentException("missing kafka.sasl.jaas.config");
118 }
119 rv.put(SaslConfigs.SASL_JAAS_CONFIG, jaasConfig);
120
121 // hardcoded values
122 rv
123 .put(
124 ConsumerConfig.KEY_DESERIALIZER_CLASS_CONFIG,
125 "org.apache.kafka.common.serialization.ByteArrayDeserializer"
126 );
127 rv
128 .put(
129 ConsumerConfig.VALUE_DESERIALIZER_CLASS_CONFIG,
130 "org.apache.kafka.common.serialization.ByteArrayDeserializer"
131 );
132 rv.put(ConsumerConfig.ENABLE_AUTO_COMMIT_CONFIG, "false");
133 rv.put(ConsumerConfig.RECEIVE_BUFFER_CONFIG, 65536);
134 rv.put(ConsumerConfig.MAX_POLL_INTERVAL_MS_CONFIG, 86400000); // avoid interrupt
135
136 return rv;
137 }

139 private Map<String, Object> parseKafkaDriverOpts(Map<String, String> opts) {
140 // driver specific options
141 Map<String, Object> rv = parseKafkaCommonOpts(opts);
142 rv.put(ConsumerConfig.GROUP_ID_CONFIG, consumerGroupId + "-driver");
143 rv.put(ConsumerConfig.AUTO_OFFSET_RESET_CONFIG, "earliest");
144 rv.put(ConsumerConfig.MAX_POLL_RECORDS_CONFIG, 1); // driver should not
145
146 return rv;
147 }

149 private Map<String, Object> parseKafkaExecutorOpts(Map<String, String> opts) {
150 // executor specific options
151 Map<String, Object> rv = parseKafkaCommonOpts(opts);
152
153 rv.put(ConsumerConfig.GROUP_ID_CONFIG, consumerGroupId + "-executor");
154 rv.put(ConsumerConfig.AUTO_OFFSET_RESET_CONFIG, "earliest");
155
156 String maxPollRecords = opts.getOrDefault("kafka.max.poll.records", "1000");
157 rv.put(ConsumerConfig.MAX_POLL_RECORDS_CONFIG, Integer.parseInt(maxPollRecords));
158
159 String fetchMaxBytes = opts.getOrDefault("kafka.fetch.max.bytes", "1024");
160 rv.put(ConsumerConfig.FETCH_MAX_BYTES_CONFIG, Integer.parseInt(fetchMaxBytes));
161
162 String fetchMaxWait = opts.getOrDefault("kafka.fetch.max.wait.ms", String.valueOf(120 * 1000));
163 rv.put(ConsumerConfig.FETCH_MAX_WAIT_MS_CONFIG, Integer.parseInt(fetchMaxWait));
164
165 String fetchMaxPartitionBytes = opts.getOrDefault("kafka.max.partition.fetch.bytes", "1024");
166 rv.put(ConsumerConfig.MAX_PARTITION_FETCH_BYTES_CONFIG, fetchMaxPartitionBytes);
167
168 return rv;
169 }

171 private Map<String, String> parseKafkaExecutorConfig(Map<String, String> opts) {
172 // executor specific config
173 Map<String, String> rv = new HashMap<>();
174 // if running in unit-test mode
175 String useMockKafkaConsumer = opts.getOrDefault("kafka.useMockKafkaConsumer", "false");
176 rv.put("useMockKafkaConsumer", useMockKafkaConsumer);
177
178 // cut-off epoch
179 String inludeEpochAndAfterString = opts
180 .getOrDefault("kafka.includeEpochAndAfter", String.valueOf(Long.MIN_VALUE / (1000 * 1000)));
181 rv.put("includeEpochAndAfter", inludeEpochAndAfterString);
182
183 return rv;
184 }
