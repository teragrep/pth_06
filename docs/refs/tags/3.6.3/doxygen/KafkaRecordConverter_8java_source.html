111 private long rfc3339ToEpoch(ZonedDateTime zonedDateTime) {
112 final Instant instant = zonedDateTime.toInstant();
113
114 final long MICROS_PER_SECOND = 1000L * 1000L;
115 final long NANOS_PER_MICROS = 1000L;
116 final long sec = Math.multiplyExact(instant.getEpochSecond(), MICROS_PER_SECOND);
117
118 return Math.addExact(sec, instant.getNano() / NANOS_PER_MICROS);
119 }

121 public UnsafeRow convert(InputStream inputStream, String partition, long offset) {
122 rfc5424Frame.load(inputStream);
123 try {
124 rfc5424Frame.next();
125 }
126 catch (IOException ioException) {
127 throw new UncheckedIOException(ioException);
128 }
129
130 final long epochMicros = rfc3339ToEpoch(new RFC5424Timestamp(rfc5424Frame.timestamp).toZonedDateTime());
131
132 // input
133 final byte[] source = eventToSource();
134
135 // origin
136 final byte[] origin = eventToOrigin();
137
138 /*
139 return RowFactory.create(
140 Timestamp.from(instant), // 0 "_time", DataTypes.TimestampType
141 UTF8String.fromBytes(message).toString(), // 1 "_raw", DataTypes.StringType
142 UTF8String.fromBytes(index).toString(), // 2 "directory", DataTypes.StringType
143 UTF8String.fromBytes(sourcetype).toString(),// 3 "stream", DataTypes.StringType
144 UTF8String.fromBytes(hostname).toString(), // 4 "host", DataTypes.StringType,
145 UTF8String.fromBytes(input).toString(), // 5 "input", DataTypes.StringType
146 partition, // 6 "partition", DataTypes.StringType
147 offset, // 7 "offset", DataTypes.LongType
148 UTF8String.fromBytes(origin).toString() // 8 "origin", DataTypes.StringType
149 );
150
151 */
152
153 rowWriter.reset();
154 rowWriter.zeroOutNullBytes();
155 rowWriter.write(0, epochMicros);
156 rowWriter.write(1, UTF8String.fromBytes(rfc5424Frame.msg.toBytes()));
157 rowWriter.write(2, UTF8String.fromBytes(rfc5424Frame.structuredData.getValue(teragrepDirectory).toBytes()));
158 rowWriter.write(3, UTF8String.fromBytes(rfc5424Frame.structuredData.getValue(teragrepStreamName).toBytes()));
159 rowWriter.write(4, UTF8String.fromBytes(rfc5424Frame.hostname.toBytes()));
160 rowWriter.write(5, UTF8String.fromBytes(source));
161 rowWriter.write(6, UTF8String.fromBytes(partition.getBytes(StandardCharsets.UTF_8)));
162 rowWriter.write(7, offset);
163 rowWriter.write(8, UTF8String.fromBytes(origin));
164
165 return rowWriter.getRow();
166 }

168 private byte[] eventToOrigin() {
169 byte[] origin;
170 Fragment originFragment = rfc5424Frame.structuredData.getValue(originHostname);
171 if (!originFragment.isStub) {

175 origin = new byte[] {};

180 private byte[] eventToSource() {
181 //input is produced from SD element event_node_source@48577 by
182 // concatenating "source_module:hostname:source". in case
183 //if event_node_source@48577 is not available use event_node_relay@48577.
184 //If neither are present, use null value.
185
186 sourceConcatenationBuffer.clear();
187
188 Fragment sourceModuleFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceSourceModule);
189 if (sourceModuleFragment.isStub) {
190 sourceModuleFragment = rfc5424Frame.structuredData.getValue(eventNodeRelaySourceModule);
191 }
192
193 byte[] source_module;
194 if (!sourceModuleFragment.isStub) {
195 source_module = sourceModuleFragment.toBytes();
196 }
197 else {
198 source_module = new byte[] {};
199 }
200
201 Fragment sourceHostnameFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceHostname);
202 if (sourceHostnameFragment.isStub) {
203 sourceHostnameFragment = rfc5424Frame.structuredData.getValue(eventNodeRelayHostname);
204 }
205
206 byte[] source_hostname;
207 if (!sourceHostnameFragment.isStub) {
208 source_hostname = sourceHostnameFragment.toBytes();
209 }
210 else {
211 source_hostname = new byte[] {};
212 }
213
214 Fragment sourceSourceFragment = rfc5424Frame.structuredData.getValue(eventNodeSourceSource);
215 if (sourceHostnameFragment.isStub) {
216 sourceSourceFragment = rfc5424Frame.structuredData.getValue(eventNodeRelaySource);
217 }
218
219 byte[] source_source;
220 if (!sourceSourceFragment.isStub) {
221 source_source = sourceSourceFragment.toBytes();
222 }
223 else {
224 source_source = new byte[] {};
225 }
226
227 // source_module:hostname:source"
228 sourceConcatenationBuffer.put(source_module);
229 sourceConcatenationBuffer.put((byte) ':');
230 sourceConcatenationBuffer.put(source_hostname);
231 sourceConcatenationBuffer.put((byte) ':');
232 sourceConcatenationBuffer.put(source_source);
233
234 sourceConcatenationBuffer.flip();
235 byte[] input = new byte[sourceConcatenationBuffer.remaining()];
236 sourceConcatenationBuffer.get(input);
237
238 return input;
239 }
