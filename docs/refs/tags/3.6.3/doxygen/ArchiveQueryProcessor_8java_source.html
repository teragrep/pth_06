187 @Override
188 public Long incrementAndGetLatestOffset() {
189 // Get initial offset as latest offset in the beginning
190 // and load that day's events into sliceTable
191 if (this.latestOffset == null) {
192 this.latestOffset = getInitialOffset();
193 sdc.pullToSliceTable(Date.valueOf(rollingDay));
194 }
195
196 // Initialize the batchSizeLimit object to split the data into appropriate sized batches
197 BatchSizeLimit batchSizeLimit = new BatchSizeLimit(quantumLength * numPartitions, totalObjectCountLimit);
198
199 // Keep loading more offsets until the limit is reached
200 while (!batchSizeLimit.isOverLimit()) {
201 WeightedOffset weightedOffsetOfNextHour = sdc.getNextHourAndSizeFromSliceTable(this.latestOffset);
202
203 // SliceTable was empty on latestOffset day
204 if (weightedOffsetOfNextHour.isStub) {
205 LOGGER.debug("Weighted offset was empty on day {}", rollingDay);
206
207 // Stop loop on endDay: no future date processing
208 if (rollingDay.isEqual(endDay) || rollingDay.isAfter(endDay)) {
209 break;
210 }
211
212 // Pull next day to sliceTable
213 rollingDay = rollingDay.plusDays(1);
214 seekToResults();
215 weightedOffsetOfNextHour = sdc.getNextHourAndSizeFromSliceTable(this.latestOffset);
216 }
217
218 // If SliceTable was not empty on the processed day, estimate the weight of it to calculate
219 // the limit and update the latestOffset.
220 if (!weightedOffsetOfNextHour.isStub) {
221 batchSizeLimit.add(weightedOffsetOfNextHour.estimateWeight(compressionRatio, processingSpeed));
222 this.latestOffset = weightedOffsetOfNextHour.offset();
223 }
224 }
225
226 LOGGER.info("Return latest offset: {}", this.latestOffset);
227 return this.latestOffset;
228 }
