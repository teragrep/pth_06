191 private void writeFilter(final String tableName, final int filterId) throws SQLException {
192 conn.prepareStatement("CREATE SCHEMA IF NOT EXISTS BLOOMDB").execute();
193 conn.prepareStatement("USE BLOOMDB").execute();
194 final String sql = "INSERT INTO `" + tableName + "` (`partition_id`, `filter_type_id`, `filter`) "
195 + "VALUES (?, (SELECT `id` FROM `filtertype` WHERE id=?), ?)";
196 final PreparedStatement stmt = conn.prepareStatement(sql);
197 final BloomFilter filter = BloomFilter.create(1000, 0.01);
198 final ByteArrayOutputStream filterBAOS = new ByteArrayOutputStream();
199 Assertions.assertDoesNotThrow(() -> {
200 filter.writeTo(filterBAOS);
201 filterBAOS.close();
202 });
203 stmt.setInt(1, 1);
204 stmt.setInt(2, filterId);
205 stmt.setBytes(3, filterBAOS.toByteArray());
206 stmt.executeUpdate();
207 stmt.close();
208 }
