181 @Override
182 public boolean next() {
183 boolean rv = false;
184
185 while (currentOffset < endOffset - 1) { // end exclusive
186 while (!kafkaRecordsIterator.hasNext()) {
187 // still need to consume more, infinitely loop because connection problems may cause return of an empty iterator
188 try {
189 ConsumerRecords<byte[], byte[]> kafkaRecords = kafkaConsumer.poll(Duration.ofSeconds(60)); // TODO parametrize
190 if (kafkaRecords.isEmpty()) {
191 LOGGER.warn("kafkaRecords empty after poll, will retry.");
192 }
193 kafkaRecordsIterator = kafkaRecords.iterator();
194 LOGGER.debug("polled at currentOffset: " + currentOffset + " endOffset: " + endOffset);
195 }
196 catch (TopicAuthorizationException topicAuthorizationException) {
197 // not authorized to access this topic, bailing out
198 LOGGER.warn("Task triggered TopicAuthorizationException: " + topicAuthorizationException);
199 return false;
200 }
201 }
202
203 ConsumerRecord<byte[], byte[]> consumerRecord = kafkaRecordsIterator.next();
204 currentOffset = consumerRecord.offset(); // update current
205
206 metricRegistry.counter("BytesProcessed").inc(consumerRecord.serializedValueSize());
207 metricRegistry.meter("BytesPerSecond").mark();
208 metricRegistry.meter("RecordsPerSecond").mark();
209 final SettableGauge<Long> kafkaTsGauge = metricRegistry.gauge("LatestKafkaTimestamp");
210 kafkaTsGauge.setValue(consumerRecord.timestamp());
211
212 try {
213 currentRow = convertToRow(consumerRecord);
214 }
215 catch (IOException ioException) {
216 throw new UncheckedIOException(ioException);
217 }
218 catch (ParseException parseException) {
219 LOGGER
220 .warn(
221 "Kafka Record at offset: " + currentOffset + " produced an exception while parsing: "
222 + parseException
223 );
224 // skip
225 if (skipNonRFC5424Records) {
226 continue;
227 }
228 else {
229 throw parseException;
230 }
231
232 }
233
234 // do the time based inclusion
235 long rfc5424time = currentRow.getLong(0); // timestamp as epochMicros
236 if (rfc5424time >= includeRowsAtAndAfterEpochMicros) {
237 rv = true;
238 break;
239 }
240 }
241
242 LOGGER.debug("next rv: " + rv);
243 return rv;
244 }

246 private InternalRow convertToRow(ConsumerRecord<byte[], byte[]> consumerRecord) throws IOException {
247 dataLength = dataLength + consumerRecord.serializedValueSize();
248 // kafka provided data
249 long offset = consumerRecord.offset();
250 int partition = consumerRecord.partition();
251 String topic = consumerRecord.topic();
252 byte[] key = consumerRecord.key();
253 byte[] value = consumerRecord.value();
254
255 // process rfc5424 provided data
256 ByteArrayInputStream payload = new ByteArrayInputStream(value);
257
258 return kafkaRecordConverter.convert(payload, String.valueOf(partition), offset);
259 }
