141 clientConfiguration.setClientExecutionTimeout(0); // do not enforce total response timeout

143 clientConfiguration.setUseReaper(true); // tear down idle connections from the pool
144 clientConfiguration.setConnectionTTL(500); // tear down after 500ms from opening so that connections are rarely re-used
