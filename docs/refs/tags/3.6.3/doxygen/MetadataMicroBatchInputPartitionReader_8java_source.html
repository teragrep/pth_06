111 @Override
112 public boolean next() throws IOException {
113 if (taskObjectList.isEmpty()) {
114 return false;
115 }
116 else {
117 rowWriter.reset();
118 rowWriter.zeroOutNullBytes();
119 ArchiveS3ObjectMetadata taskObject = taskObjectList.removeFirst();
120
121 // Use metadata java object to easily form a json representation of metadata
122 final String rawColumn = new Gson()
123 .toJson(new Metadata(taskObject.uncompressedSize, taskObject.compressedSize));
124 // use logtimeEpoch as _time
125 rowWriter
126 .write(0, rfc3339ToEpoch(Instant.ofEpochSecond(taskObject.logtimeEpoch).atZone(ZoneId.systemDefault())));
127 rowWriter.write(1, UTF8String.fromString(rawColumn));
128 rowWriter.write(2, UTF8String.fromString(taskObject.directory));
129 rowWriter.write(3, UTF8String.fromString(taskObject.stream));
130 rowWriter.write(4, UTF8String.fromString(taskObject.host));
131 rowWriter.write(5, UTF8String.fromString(""));
132 rowWriter.write(6, UTF8String.fromString(taskObject.id));
133 rowWriter.write(7, currentOffset++);
134 rowWriter.write(8, UTF8String.fromString(""));
135 return true;
136 }

159 static long rfc3339ToEpoch(ZonedDateTime zonedDateTime) {
160 final Instant instant = zonedDateTime.toInstant();
161
162 final long MICROS_PER_SECOND = 1000L * 1000L;
163 final long NANOS_PER_MICROS = 1000L;
164 final long sec = Math.multiplyExact(instant.getEpochSecond(), MICROS_PER_SECOND);
165
166 return Math.addExact(sec, instant.getNano() / NANOS_PER_MICROS);
167 }
