75 public static String stripQuotes(String value) {
76 Matcher m = Pattern.compile("^\"(.*)\"$").matcher(value);
77 Matcher m1 = Pattern.compile("^'(.*)'$").matcher(value);
78
79 String strUnquoted = value;
80 // check "-quotes
81 if (m.find()) {
82 strUnquoted = m.group(1);
83 }
84 else {
85 // check '-quotes
86 if (m1.find()) {
87 strUnquoted = m1.group(1);
88 }
89 }
90 return strUnquoted;
91 }

93 public static String addQuotes(String s) {
94 String str = s;
95 // If quotes are missing, add them
96 if (!s.startsWith("\"") && !s.startsWith("'")) {
97 str = "\"".concat(s);
98 }
99 if (!s.endsWith("\"") && !s.endsWith("'")) {
100 str = str.concat("\"");
101 }
102 return str;

109 public static long relativeTimeModifier(Timestamp timestamp, String value) {
110 Instant rv = null;
111 Matcher matcher = Pattern.compile("^-?\\d+").matcher(value);
112
113 if (!matcher.find()) {
114 throw new NumberFormatException("Unknown relative time modifier string [" + value + "]");
115 }
116
117 long v = Long.parseLong(matcher.group());
118 matcher.group().length();
119 String unit = value.substring(matcher.group().length(), value.length());
120
121 LocalDateTime now = timestamp.toLocalDateTime();
122 rv = timestamp.toInstant();
123 if (
124 "s".equalsIgnoreCase(unit) || "sec".equalsIgnoreCase(unit) || "secs".equalsIgnoreCase(unit)
125 || "second".equalsIgnoreCase(unit) || "seconds".equalsIgnoreCase(unit)
126 ) {
127 rv = rv.plusSeconds(v);
128 }
129 if (
130 "m".equalsIgnoreCase(unit) || "min".equalsIgnoreCase(unit) || "minute".equalsIgnoreCase(unit)
131 || "minutes".equalsIgnoreCase(unit)
132 ) {
133 rv = rv.plus(v, ChronoUnit.MINUTES);
134 }
135 if (
136 "h".equalsIgnoreCase(unit) || "hr".equalsIgnoreCase(unit) || "hrs".equalsIgnoreCase(unit)
137 || "hour".equalsIgnoreCase(unit) || "hours".equalsIgnoreCase(unit)
138 ) {
139 rv = rv.plus(v, ChronoUnit.HOURS);
140 }
141 if ("d".equalsIgnoreCase(unit) || "day".equalsIgnoreCase(unit) || "days".equalsIgnoreCase(unit)) {
142 rv = rv.plus(v, ChronoUnit.DAYS);
143 }
144 // longer times are not supported with Instant, so we need to change it into the
145 // localdatetime object
146 if ("w".equalsIgnoreCase(unit) || "week".equalsIgnoreCase(unit) || "weeks".equalsIgnoreCase(unit)) {
147 now = now.plusWeeks(v);
148 rv = now.toInstant(ZoneOffset.UTC);
149 }
150 if ("mon".equalsIgnoreCase(unit) || "month".equalsIgnoreCase(unit) || "months".equalsIgnoreCase(unit)) {
151 now = now.plusMonths(v);
152 rv = now.toInstant(ZoneOffset.UTC);
153 }
154 if (
155 "y".equalsIgnoreCase(unit) || "yr".equalsIgnoreCase(unit) || "yrs".equalsIgnoreCase(unit)
156 || "year".equalsIgnoreCase(unit) || "years".equalsIgnoreCase(unit)
157 ) {
158 now = now.plusYears(v);
159 rv = now.toInstant(ZoneOffset.UTC);
160 }
161 // System.out.println("Relative time (UTC)=" + rv.atZone(ZoneOffset.UTC) + "
162 // unixtime=" + rv.getEpochSecond());
163 return rv.getEpochSecond();
164 }

166 public static long unixEpochFromString(String timevalue, String tf) {
167 long rv = 0;
168 // Remove '"' around string if needed
169 timevalue = stripQuotes(timevalue);
170 try {
171 if (tf != null && !tf.isEmpty()) {
172 if ("%s".equals(tf)) {
173 // unix-epoch string so
174 long epoch = Long.parseLong(timevalue);
175 rv = epoch;
176 }
177 else {
178 // Not yet implemented so use default
179 LocalDateTime time = LocalDateTime
180 .parse(timevalue, DateTimeFormatter.ofPattern("MM/dd/yyyy:HH:mm:ss"));
181 final ZonedDateTime zonedtime = ZonedDateTime.of(time, ZoneId.systemDefault()); // .atZone(fromZone);
182 final ZonedDateTime converted = zonedtime.withZoneSameInstant(ZoneOffset.UTC);
183 rv = converted.toEpochSecond();
184 //System.out.println("LocalDatetime="+time+ " TZ="+ZoneId.systemDefault());
185 //System.out.println("Datetime(UTC)="+converted+" timestamp="+rv);
186 }
187 }
188 else {
189 // Use default format
190 LocalDateTime time = LocalDateTime.parse(timevalue, DateTimeFormatter.ofPattern("MM/dd/yyyy:HH:mm:ss"));
191 final ZonedDateTime zonedtime = ZonedDateTime.of(time, ZoneId.systemDefault()); // .atZone(fromZone);
192 final ZonedDateTime converted = zonedtime.withZoneSameInstant(ZoneOffset.UTC);
193 rv = converted.toEpochSecond();
194 //System.out.println("LocalDatetime="+time+ " TZ="+ZoneId.systemDefault());
195 //System.out.println("Datetime(UTC)="+converted+" timestamp="+rv);
196 }
197 }
198 catch (NumberFormatException exn) {
199 exn.printStackTrace();
200 throw new RuntimeException("TimeQualifier conversion error: <" + timevalue + "> can't be parsed.");
201 }
202 catch (DateTimeParseException ex) {
203 ex.printStackTrace();
204 throw new RuntimeException("TimeQualifier conversion error: <" + timevalue + "> can't be parsed.");
205 }
206 return rv;
207 }

210 public static String elementAsString(Element el) {
211 String str = null;
212 try {
213 TransformerFactory transFactory = TransformerFactory.newInstance();
214 Transformer transformer = transFactory.newTransformer();
215 StringWriter buffer = new StringWriter();
216 transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
217 transformer.transform(new DOMSource(el), new StreamResult(buffer));
218 str = buffer.toString();
219 }
220 catch (TransformerConfigurationException tex) {
221 if (LOGGER.isDebugEnabled()) {
222 LOGGER.debug(tex.getMessage(), tex);
223 }
224 }
225 catch (TransformerException ex) {
226 if (LOGGER.isDebugEnabled()) {
227 LOGGER.debug(ex.getMessage());
228 }
229 }
230 return str;
231 }
