58public final class BatchTaskQueue implements Comparable<BatchTaskQueue> {
59
60 private final double compressionRatio = 15.5F;
61 private final double contextSwitchCost = 0.1F; // seconds
62 private final double processingSpeed = 273 / 2F; // rlo_06 273 megabytes per second, spark the half of it
63
64 private final List<BatchUnit> queue;
65 private double queueTime = 0L; // seconds how long the queue will take to process
66
67 BatchTaskQueue() {
68 this.queue = new ArrayList<>();
69 }
70
71 // give estimate on the queueTime after adding an object
72 private double estimate(BatchUnit batchUnit) {
73 return queueTime + (batchUnit.getSize() * compressionRatio) / 1024 / 1024 / processingSpeed;
74 }
75
76 public void add(BatchUnit batchUnit) {
77 queue.add(batchUnit);
78 queueTime = estimate(batchUnit);
79 }
80
81 public List<BatchUnit> getQueue() {

83 }

85 public double getQueueTime() {

87 }
