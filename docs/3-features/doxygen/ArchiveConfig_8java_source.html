75 public ArchiveConfig(Map<String, String> opts) {
76 s3EndPoint = getOrThrow(opts, "S3endPoint");
77 s3Identity = getOrThrow(opts, "S3identity");
78 s3Credential = getOrThrow(opts, "S3credential");
79 dbUsername = getOrThrow(opts, "DBusername");
80 dbPassword = getOrThrow(opts, "DBpassword");
81 dbUrl = getOrThrow(opts, "DBurl");
82 bloomEnabled = opts.getOrDefault("bloom.enabled", "false").equalsIgnoreCase("true");
83 withoutFilters = opts.getOrDefault("bloom.withoutFilters", "false").equalsIgnoreCase("true");
84 withoutFiltersPattern = opts.getOrDefault("bloom.withoutFiltersPattern", "");
85 bloomDbName = opts.getOrDefault("DBbloomdbname", "bloomdb");
86
87 dbJournalDbName = opts.getOrDefault("DBjournaldbname", "journaldb");
88 dbStreamDbName = opts.getOrDefault("DBstreamdbname", "streamdb");
89
90 // skip not rfc5424 parseable files
91 skipNonRFC5424Files = opts.getOrDefault("skipNonRFC5424Files", "false").equalsIgnoreCase("true");
92
93 archiveIncludeBeforeEpoch = Long
94 .parseLong(opts.getOrDefault("archive.includeBeforeEpoch", String.valueOf(Long.MAX_VALUE)));

96 isStub = false;
